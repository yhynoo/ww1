on_actions = {
    on_peace = {
        effect = {
            if = {
                limit = { 
                    or = { tag = UBD tag = UKR }
                    not = { has_rule = can_join_factions }
                }
                set_rule = { can_join_factions = yes }
            }
        }
    }

    on_war_relation_added = {
        effect = {
            if = {
                limit = { original_tag = FROM }

                # countries in civil war can't join factions
                set_rule = { can_join_factions = no }
                FROM = { 
                    set_rule = { can_join_factions = no }
                    add_popularity = { ideology = ROOT popularity = -0.25 }
                }

                # remove advisors and ideas
                _manage_advisors_and_ideas = yes
                FROM = { _manage_advisors_and_ideas = yes }

                # can't magically peace out
                every_country = {
                    limit = { 
                        has_war_with = ROOT
                        or = { is_faction_leader = yes is_in_faction = no }
                    }
                    FROM = {
                        declare_war_on = { target = PREV type = topple_government }
                    }
                }

                every_country = {
                    limit = { 
                        has_war_with = FROM
                        or = { is_faction_leader = yes is_in_faction = no }
                    }
                    ROOT = { declare_war_on = { target = PREV type = topple_government } }
                }
            }
        }
    }

    on_civil_war_end = {
        effect = {
            set_rule = { can_join_factions = yes }

            if = {
                limit = { FROM = { original_tag = GER has_government = communism } }
                news_event = { id = germany_liebknecht.1 days = 7 }
            }

            # ---

            if = {
                limit = { has_cosmetic_tag = generic_communism }
                drop_cosmetic_tag = yes
            }   
        }
    }
}
