on_actions = {
    on_war_relation_added = {
        effect = {
            if = {
                limit = { original_tag = FROM }

                # countries in civil war can't join factions
                set_rule = { can_join_factions = no }
                FROM = { 
                    set_rule = { can_join_factions = no }
                    add_popularity = { ideology = ROOT popularity = -0.25 }
                }

                # can't magically peace out
                every_country = {
                    limit = { 
                        has_war_with = ROOT
                        or = { is_faction_leader = yes is_in_faction = no }
                    }
                    FROM = {
                        declare_war_on = { target = PREV type = topple_government }
                    }
                }

                every_country = {
                    limit = { 
                        has_war_with = FROM
                        or = { is_faction_leader = yes is_in_faction = no }
                    }
                    ROOT = { declare_war_on = { target = PREV type = topple_government } }
                }
            }
        }
    }

    on_civil_war_end = {
        effect = {
            set_rule = { can_join_factions = yes }
        }
    }
}
