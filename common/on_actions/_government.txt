on_actions = {
    on_government_change = {
        effect = { 
            if = {
                limit = { 
                    has_government = communism
                    not = { or = { tag = MEX tag = SOV } }
                }

                # sets flag and country color
                set_cosmetic_tag = generic_communism
            }
        }
    }

    on_ruling_party_change = {
        effect = {
            if = {
                limit = {
                    tag = SOV
                    old_ideology_token = monarchism
                }
            }

            news_event = { id = soviet_tsar.1 hours = 6 }
        }
    }
}
