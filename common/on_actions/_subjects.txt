on_actions = {
    on_subject_autonomy_level_change = {
        effect = {
            if = {
                limit = { 
                    tag = RAJ
                    is_subject = no
                }
                drop_cosmetic_tag = yes
            }

            if = {
                limit = { 
                    tag = UGA
                    is_subject = no
                }

                drop_cosmetic_tag = yes
                news_event = { id = uganda_autonomy.3 hours = 6 }
            }

            # polish monarchs
            if = {
                limit = { 
                    tag = POL 
                    is_subject = yes 
                }

                if = { 
                    limit = { is_subject_of = AUS }
                    promote_character = POL_karl_stephan_von_habsburg
                }

                if = { 
                    limit = { is_subject_of = GER }
                    promote_character = POL_friedrich_christian_wettin
                }
            }
        }
    }
}
