on_actions = {
    on_war = {
        effect = {
            if = {
                limit = { has_civil_war = no }

                # pre-launch the stalemate event
                country_event = { id = _armistice.2 days = 180 random_days = 30 }
            }

            # adding the side of the great war
            if = {
                limit = {
                    not = { has_country_flag = side_of_the_great_war }
                    any_allied_country = { has_country_flag = side_of_the_great_war }
                }
                set_country_flag = side_of_the_great_war
            }

        }
    }

    on_annex = {
        effect = {
            if = {
                limit = {   
                    tag = TUR     
                    FROM = { tag = GRE }
                }
                add_country_flag = we_beat_the_greeks
            }

            if = {
                limit = {   
                    tag = TUR     
                    FROM = { tag = SER }
                }
                add_country_flag = we_beat_the_serbs
            }
        }
    }
}
