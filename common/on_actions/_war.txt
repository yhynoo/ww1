on_actions = {
    on_war_relation_added = {
        effect = {
            if = {
                limit = { ROOT = { original_tag = FROM } }      # same tags: civil war.
                
                FROM = { 
                    add_popularity = { ideology = FROM popularity = 0.15 }
                    add_popularity = { ideology = ROOT popularity = -0.4 } 
                }

                every_country = {
                    limit = { original_tag = ROOT has_civil_war = yes }
                    setup_civil_war = yes
                }
            }
        }
    }

    on_civil_war_end = {
        effect = {
            set_rule = { can_join_factions = yes }              # can join factions again.
            if = { limit = { not = { has_country_flag = after_civil_war } } set_country_flag = after_civil_war }

            # event
            news_event = { id = unrest.15 hours = 6 }
        }
    }

    on_peace = {
        effect = {
            if = {
                limit = { has_country_flag = during_uprising exists = yes }
                news_event = other.3
            }
        }
    }

    on_annex = {
        effect = {
            # chinese warlords
            if = {
                limit = {
                    FROM = {
                        or = { tag = EAT tag = FEN tag = GUA tag = XMA tag = ANH tag = SHA tag = YUN }
                    }
                }
                FROM = { set_country_flag = warlord_defeated }
            }
        }
    }
}
