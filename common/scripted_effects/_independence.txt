### UPRISING TROOPS SETUP HELPERS ###

setup_uprising_country = {
    every_controlled_state = {
        limit = { is_core_of = PREV.PREV impassable = no }
        set_state_owner_to = PREV.PREV

        random = {
            chance = 50
            setup_uprising_state = yes
        }
    }
}

setup_uprising_state = {
    if = {
        limit = { not = { is_controlled_by = PREV.PREV } }
        transfer_state_to = PREV.PREV
    }

    create_unit = {
        division = "name = \"Uprising Troops\" division_template = \"Uprising Troops\" start_experience_factor = 0.2" count = 2
        owner = PREV.PREV
    }
}

setup_uprising_tribal_units = {
    division_template = {
        name = "Tribal Troops"
        regiments = {
            camelry = { x = 0 y = 0 }      camelry = { x = 1 y = 0 }
            camelry = { x = 0 y = 1 }      camelry = { x = 1 y = 1 }
        }
    }

    every_controlled_state = {
        limit = { impassable = no }
        random = {
            chance = 50
            create_unit = {
                division = "name = \"Tribal Troops\" division_template = \"Tribal Troops\" start_experience_factor = 0.2" count = 1
                owner = PREV
            }
        }
    }
}

setup_uprising_troops = {
    if = {
        limit = { not = { has_template = "Uprising Troops" } }
        division_template = {
            name = "Uprising Troops"
            regiments = {
                infantry = { x = 0 y = 0 }      infantry = { x = 1 y = 0 }
                infantry = { x = 0 y = 1 }      infantry = { x = 1 y = 1 }
            }
        }
    }
}
