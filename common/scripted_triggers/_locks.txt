civil_war_industry_lock = {
    if = { 
        limit = { has_country_flag = during_civil_war }
        if = {
            limit = { not = { has_idea = idea_eroded_state } }
            custom_trigger_tooltip = {
                tooltip = tt_civil_war_lock
                and = { has_country_flag = during_civil_war has_completed_focus = XXX_wheat_and_lead }
            }
        }
        else = { 
            custom_trigger_tooltip = { tooltip = tt_eroded_state    always = no }
        }
    }
}
