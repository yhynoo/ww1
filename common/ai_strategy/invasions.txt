## TARGETS

invasions_on_england = {
    enable = { 
        has_war_with = ENG
        or = {
            not = { has_war_with = FRA }
            FRA = { has_capitulated = yes }
        }
    }
    abort_when_not_enabled = yes

    ai_strategy = {
        type = invasion_unit_request
        state_trigger = { 
            or = { state = 125 state = 126 state = 127 }
            is_controlled_by = ENG
        }
        value = 60
    }
}
