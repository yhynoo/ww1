dont_naval_invade = {
    allowed = {
        not = {
            tag = AST tag = ENG tag = JAP tag = USA
        }
    }
    enable = { always = yes }
    abort_when_not_enabled = yes

    ai_strategy = {
        type = invasion_unit_request
        country_trigger = { always = yes }
        value = -100
    }
}

# japan can invade tsingtao and a few islands only
entente_tsingtao = {
    allowed = { tag = JAP }
    enable = { always = yes }
    abort_when_not_enabled = yes

    ai_strategy = {
        type = invasion_unit_request
        state_trigger = { 
            not = {
                is_controlled_by = GER
                or = { is_on_continent = asia is_on_continent = australia }
            }
        }
        value = -100
    }
}
