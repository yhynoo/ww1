# DIPLOMACY

SOV_alliances = {
	allowed = { original_tag = SOV }
	enable = { 
		not = { has_government = communism }
		or = { is_guaranteed_by = FRA is_in_faction_with = FRA } 
	}
	abort_when_not_enabled = yes

	# alliances
	ai_strategy = { type = alliance		id = FRA		value = 200 }

	# aggressive wargoals
	ai_strategy = { type = conquer		id = AUS		value = 40 }
	ai_strategy = { type = conquer		id = TUR		value = 20 }
}

SOV_communist_expansion = {
	allowed = { original_tag = SOV }
	enable = { 
		or = { 
			and = {
				or = { tag = ARM tag = POL tag = TMS tag = UBD tag = UKR }
				is_in_faction = no
			}
			any_owned_state = { is_core_of = SOV }
		} 
	}
	
	reversed = yes
	enable_reverse = { 
		has_government = communism 
		has_civil_war = no
	}

	# aggressive wargoals
	ai_strategy = { type = conquer		id = SOV		value = 100 }
	ai_strategy = { type = declare_war	id = SOV		value = 100 }
}
