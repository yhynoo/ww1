# greater fighter production during the war
war_air_production = {
	allowed = {
		and = { has_war = yes has_civil_war = no }
		not = { num_of_military_factories < 15 }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = equipment_production_factor
		id = fighter
		value = 60
	}

	ai_strategy = {
		type = equipment_production_factor
		id = cas
		value = 15
	}
 }
