defend_allies = {
	enable = { always = yes }
	ai_strategy = {
		type = front_unit_request		country_trigger = { or = { has_guaranteed = FROM is_guaranteed_by = FROM } }
		value = -600
	}
}

defend_homeland = {
	enable = { always = yes }
	ai_strategy = {
		type = front_unit_request		state_trigger = { owner = FROM.FROM 	is_in_home_area = yes 	not = { is_fully_controlled_by = FROM.FROM } }
		value = 200
	}
}

defend_if_abgenutzt_or_smol = {
	enable = { 
		or = {
			and = { num_divisions < 12 	has_country_flag = side_of_the_great_war }
			
			has_manpower < 20000
			stockpile_ratio = {	archetype = infantry_equipment ratio < 0.2 }
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control	country_trigger = { always = yes }		execution = careful
	}
}
