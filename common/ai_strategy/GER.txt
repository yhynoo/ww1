# germany extra militarization
germany_army_production = {
	allowed = { original_tag = GER }
	enable = { always = yes }
	abort = { always = no }

	# prepare more units
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = 45
	}

	# focus on military factories
	ai_strategy = {
		type = added_military_to_civilian_factory_ratio
		value = 15
	}

	# also, build airships
	ai_strategy = {
		type = unit_ratio
		id = strategic_bomber
		value = 20
	}

	ai_strategy = {
		type = equipment_production_factor
		id = strategic_bomber
		value = 20
	}

	# and submarines
	ai_strategy = {
		type = unit_ratio
		id = submarine
		value = 120
	}
}

germany_focuses_on_france = {
	allowed = { original_tag = GER }
	enable = { 
		has_war_with = FRA
		FRA = { has_capitulated = no }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_unit_request
		tag = FRA
		value = 200
	}
}

# don't waste too much stuff on africa
germany_avoid_africa = {

	ai_strategy = {

		value = -100
	}
