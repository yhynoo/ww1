# DIPLOMACY

FRA_alliances = {
    allowed = { original_tag = FRA }
	enable = { and = { is_faction_leader = yes has_government = democratic } }
	abort_when_not_enabled = yes

	# alliances
	ai_strategy = { type = alliance		id = ENG		value = 40 }
	
	if = {
		limit = { has_guarenteed = SOV }
		ai_strategy = { type = alliance		id = SOV		value = 100 }
	}

	# aggressive wargoals in africa
	ai_strategy = { type = conquer		id = ALG		value = 100 }
	ai_strategy = { type = conquer		id = MOR		value = 100 }
}

FRA_attention = {
    allowed = { original_tag = FRA }
    enable = { not = { or = { is_ally_with = ENG is_ally_with = SOV } } }
    abort_when_not_enabled = yes

    ai_strategy = { type = avoid_starting_wars      value = -100 }
}
