## WAR ADJUSTMENTS

war_mobilization = {
	allowed = { always = yes }
	enable = { 
		or = {
			threat > 0.1
			and = { has_war = yes has_civil_war = no }
		}
	}
	abort_when_not_enabled = yes

	# add infantry
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = 45
	}

	# make more guns
	ai_strategy = {
		type = added_military_to_civilian_factory_ratio
		value = 30
	}

	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = 30
	}
}

civil_war_adjustment = {
	allowed = { always = yes }
	enable = { has_civil_war = yes }
	abort_when_not_enabled = yes

	# drop infantry, make cheaper units instead
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = -30
	}

	ai_strategy = {
		type = role_ratio
		id = garrison
		value = 60
	}
}
