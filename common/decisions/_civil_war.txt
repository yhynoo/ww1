        visible = {
            has_civil_war = no
            any_neighbor_country = { has_civil_war = yes }
        }

        icon = generic_industry
        cost = 50
        days_remove = 90

        targets_dynamic = yes
        target_trigger = { 
            FROM = {
                has_civil_war = yes
                is_neighbor_of = ROOT
                not = { has_war_with = ROOT }
            }
        }

            base = 0
            modifier = { and = { has_war = no FROM = { has_government = ROOT } } add = 6 }

            # Americans help Mexico
            modifier = { has_opinion_modifier = opinion_american_arms factor = 2 }
        }

        available = {
            has_equipment = { infantry_equipment > 599 }
        }

        complete_effect = {
            add_war_support = 0.025
            send_equipment = { equipment = infantry_equipment amount = 600 target = FROM }
        }
    }
    generic_truce = {
        visible = { 
            has_civil_war = yes
        }

        icon = border_war
        cost = 75
        days_re_enable = 90

        targets_dynamic = yes
        target_trigger = {
            FROM = {
                has_war_with = ROOT

                or = { is_faction_leader = yes is_in_faction = no }
                not = {
                    any_country = { original_tag = ROOT is_ally_with = FROM }       # we won't make peace with the allies of our civil war adversary
                }
            }
        }

        ai_will_do = { 
            base = 9
            modifier = { surrender_progress > 0.1 factor = 2 }

            # mexico won't do it:
            modifier = { original_tag = MEX add = -24 }
        }

        days_remove = 7
        remove_effect = {
            # it takes a moment before it can start.
            FROM = { country_event = { id = _offer.1 hours = 6 random_hours = 6 } }
        }

    }

        visible = { has_civil_war = no }
