            hidden_effect = {

                    random_country = {
                        limit = { original_tag = MEX has_government = communism }
                        USA = { add_to_war = { targeted_alliance = MEX enemy = PREV } }
                    }
                }
                add_to_war = { targeted_alliance = MEX enemy = COL }

                every_state = {
                    limit = { or = { state = 304 state = 312 state = 313 state = 314 state = 316 state = 317 } }
                    add_claim_by = USA
                }

                country_event = { id = usa_monroe_doctrine.4 days = 2 random_hours = 6 }
            }
        }
    }
    guarantees_for_latin_america = {
        cost = 100
        fire_only_once = yes

        ai_will_do = {
            base = 6
            modifier = { threat > 0.2 factor = 2 }
            modifier = { FRA = { has_government = monarchism } factor = 2 }
        }

        complete_effect = {
            news_event = { id = usa_monroe_doctrine.5 hours = 12 random_hours = 6 }

            if = {
                limit = { ARG = { has_government = ROOT } }
                give_guarantee = ARG
            }

            if = {
                limit = { BRA = { has_government = ROOT } }
                give_guarantee = BRA
            }

            if = {
                limit = { COL = { has_government = ROOT } }
                give_guarantee = COL
            }

            if = {
                limit = { MEX = { has_government = ROOT } }
                give_guarantee = MEX
            }

            if = {
                limit = { PRU = { has_government = ROOT } }
                give_guarantee = PRU
            }
        }
    }
    claim_greenland = {
        available = { has_government = authoritarianism }
