peace_ai_desires = {

    # TAKING STATES

    take_states = {
        peace_action_type = { take_states }

        enable = {
            ROOT = { ROOT.FROM }

            if = {
                limit = { ROOT.FROM = { has_government = democratic } }
                not = { ROOT.FROM.FROM.FROM = { is_core_of = ROOT.FROM } }          # democracies only take cores.
            }

            if = {
                limit = { ROOT.FROM = { has_government = monarchism } }
                not = {
                    ROOT.FROM.FROM.FROM = { is_core_of = ROOT.FROM }                # monarchists take cores
                    ROOT.FROM.FROM.FROM = { is_claimed_by = ROOT.FROM }             # and also claims
                }          
            }

            if = {
                limit = { ROOT.FROM = { has_government = communism } }
                not = {
                    ROOT.FROM.FROM.FROM = { is_core_of = ROOT.FROM }                # communists take cores
                    ROOT.FROM.FROM.FROM = { is_claimed_by = ROOT.FROM }             # and also claims
                    ROOT.FROM.FROM.FROM = { is_controlled_by = ROOT.FROM }          # and also whatever they control.
                }          
            }
        }

        ai_desire = -600
    }

    # PUPPET
    dont_puppet_large_countries = {
        peace_action_type = { puppet force_government }
        enable = {
            ROOT.FROM.FROM = { check_variable = { num_owned_states > 6 } }
        }
        ai_desire = -600
    }

    continuous_puppet = {
        peace_action_type = { puppet }
        enable = {
            ROOT.FROM = { pc_is_puppeted_by = ROOT }
        }
        ai_desire = 600
    }

    # LIBERATING
    liberating_is_broken = {
        peace_action_type = { liberate }
        enable = { 
            always = yes
        }
        ai_desire = -600
    }

    # GOVERNMENT
    government_change_is_broken_too = {
        peace_action_type = { force_government }
        enable = {
            always = yes
        }
        ai_desire = -600
    }
}
