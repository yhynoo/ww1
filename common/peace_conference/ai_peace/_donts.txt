peace_ai_desires = {

    # DOMINIONS

    dont_even_try = {
        peace_action_type = { puppet }
        enable = { 
            FROM = { 
                check_variable = { num_owned_states > 6 } 
            } 
        }
        ai_desire = -600
    }

    dont_puppet_and_force_government = {
        peace_action_type = { puppet }
        enable = {
            if = {
                limit = { has_government = democratic }
                FROM = { pc_is_forced_government_to = democratic }
            }
            else_if = {
                limit = { has_government = communism }
                FROM = { pc_is_forced_government_to = communism }
            }
            else_if = {
                limit = { has_government = monarchism }
                FROM = { pc_is_forced_government_to = monarchism }
            }
            else = {
                always = no
            }
        }
        ai_desire = -600
    }

    # IDEOLOGY

    dont_force_government_and_puppet = {
        peace_action_type = { force_government }
        enable = { FROM = { pc_is_puppeted_by = ROOT } }
        ai_desire = -600
    }

    # LIBERATING

    dont_liberate_if_puppeted = {
        peace_action_type = { liberate }
        enable = { FROM = { pc_is_puppeted = yes } }
        ai_desire = -600
    }

    # TAKING

    dont_take_state_if_already_puppeted_by_us = {
        peace_action_type = { take_state }
        enable = { FROM = { pc_is_puppeted_by = ROOT } }
        ai_desire = -600
    }
}
