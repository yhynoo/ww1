add_namespace = soviet_agitation

news_event = {
    id = soviet_agitation.1
    title = "Outpost Attacked in Arkhangelsk"
    desc = soviet_agitation.1.d

    trigger = { 
        tag = SOV

        or = { surrender_progress > 0.05 longest_war_length > 12 }
        not = { has_completed_focus = SOV_strength_in_unity }
        
        controls_state = 214    # we must control Arkhangelsk, otherwise it makes no sense
    }

    fire_only_once = yes

    option = {
        name = "Very disturbing."

        add_war_support = -0.05
        add_popularity = { ideology = communism popularity = 0.05 }
    }
}
