add_namespace = soviet_agitation

news_event = {
    id = soviet_agitation.1
    title = "Outpost Attacked in Arkhangelsk"
    desc = soviet_agitation.1.d

    trigger = { 
        tag = SOV

        or = { surrender_progress > 0.15 longest_war_length > 18 }
        not = { has_completed_focus = SOV_strength_in_unity }
        
        controls_state = 214    # we must control Arkhangelsk, otherwise it makes no sense
    }

    fire_only_once = yes

    option = {
        name = "Very disturbing."
        add_popularity = { ideology = communism popularity = 0.05 }
    }
}

country_event = {
    id = soviet_agitation.2
    title = "Lenin's Arrival"
    desc = soviet_agitation.2.d

    fire_only_once = yes

    trigger = { 
        tag = SOV
        has_war_with = GER
        or = { surrender_progress > 0.2 longest_war_length > 24 }

        has_civil_war = no
        not = { has_government = communism }
    }

    option = {
        name = "Don't provoke the workers."
        
        add_popularity = { ideology = communism popularity = 0.05 }
        promote_character = SOV_vladimir_lenin

        ai_chance = {
            base = 1
            modifier = { 
                or = { has_completed_focus = SOV_traditionalism surrender_progress < 0.05 }
                factor = 0.5
            }
        }
    }
    option = {
        name = "Arrest him."
        add_stability = -0.05

        ai_chance = {
            base = 1
            modifier = { has_completed_focus = SOV_adjust_the_government factor = 0 }
        }
    }
}

country_event = {
    id = soviet_agitation.3
    title = "The End of the Empire?"
    desc = soviet_agitation.3.d

    fire_only_once = yes

    trigger = { 
        tag = SOV
        not = { has_government = communism }

        has_civil_war = no
        or = { communism > 0.45 has_completed_focus = SOV_over_the_edge }
    }
    mean_time_to_happen = { days = 14 }

    option = {
        name = "All power to the councils!"
        start_civil_war = { ideology = monarchism ruling_party = communism size = 0.5 }

        # manage the war
        hidden_effect = { russia_civil_war = yes }
    }
    option = {
        name = "We must hold out."
        start_civil_war = { ideology = communism size = 0.6 }

        # manage the war
        hidden_effect = { russia_civil_war = yes }
    }
}
