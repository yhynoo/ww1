news_event = {
    id = soviet_agitation.1
    title = "Communist Agitation in the Army"
    desc = soviet_agitation.1.d

    fire_only_once = yes
    trigger = {
        tag = SOV
        not = { or = { has_civil_war = yes has_government = communism } }

        has_stability < 0.3
        or = { longest_war_length > 18 surrender_progress > 0.15 }
    }

    option = {
        name = "We must do something about this."
        add_timed_idea = { idea = idea_agitation_in_army days = 180 }

        hidden_effect = {
            news_event = { id = soviet_agitation.2 days = 28 random_days = 14 }
            GER = { country_event = { id = soviet_agitation.4 days = 42 random_days = 28 } }
        }
    }
}
