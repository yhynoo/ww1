country_event = {
    id = poland_intermarium.1
    title = "Poland demands Eastern Pomerania!"
    desc = poland_intermarium.1.d

    is_triggered_only = yes

    option = {
        name = "We must agree to this."
        news_event = { id = poland_intermarium.2 hours = 12 random_hours = 6 }

        every_state = {
            limit = {
                or = { state = 85 state = 86 state = 807 }
                not = { is_owned_by = POL }
            }
            transfer_state_to = POL
            add_core_of = POL
        }

        ai_chance = { 
            base = 1
            modifier = { has_war = yes factor = 3 }
        }
    }

    option = {
        name = "They are making an awful mistake."

        POL = {
            news_event = { id = poland_intermarium.3 hours = 12 random_hours = 6 }
            create_wargoal = {
                target = GER
                type = take_claimed_state
            }
        }

        ai_chance = { base = 3 }
    }
}
