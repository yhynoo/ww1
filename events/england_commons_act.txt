add_namespace = britain_commons_act

### 1. News: Mass Demonstrations in London
news_event = {
    id = britain_commons_act.1
    title = "Mass Demonstrations in London"
    desc = britain_commons_act.1.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "All power to the councils!"
        ENG = { country_event = { id = britain_commons_act.2 hours = 12 random_hours = 6 } }

        set_politics = {
            ruling_party = communism
            elections_allowed = no
        }

        hidden_effect = {
            add_popularity = {
                ideology = communism
                popularity = 0.1
            }
        }
    }

    option = {
        name = "Disperse the demonstration!"
        ENG = { country_event = { id = britain_commons_act.3 hours = 12 random_hours = 6 } }

        if = {
            limit = { not = { has_government = communism } }

            start_civil_war = {
                ideology = communism
                size = 0.5
            }

            hidden_effect = {
                add_popularity = {
                    ideology = communism
                    popularity = -0.25
                }
            }
        }
    }   
}
