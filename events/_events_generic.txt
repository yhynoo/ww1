## -- demands
add_namespace = _demand

country_event = {
    id = _demand.1
    title = "[FROM.GetName] Demands Territories"
    desc = _demand.1.d

    is_triggered_only = yes

    option = {
        name = "We can do nothing about it."
        FROM = { news_event = { id = _demand.2 hours = 6 } }

        every_owned_state = {
            limit = { or = { is_core_of = FROM is_claimed_by = FROM } }
            FROM = { set_state_owner = PREV }
        }

        ai_chance = { 
            factor = 1 
            modifier = { strength_ratio = { tag = FROM ratio > 2.0 } factor = 2 }
        }
    }
    option = {
        name = "No!"
        FROM = { 
            create_wargoal = { type = take_claimed_state target = ROOT } 
            add_named_threat = { threat = 4 name = "[ROOT.GetName] Dismisses Our Claims!" }
        }
        
        ai_chance = { 
            factor = 2
            modifier = { any_country = { or = { is_ally_with = ROOT has_guaranteed = ROOT } } factor = 2 }
        }
        
    }
}

## -- war
add_namespace = _war

news_event = {
    id = _war.0         # this is just the trigger event.
    hidden = yes

    trigger = {
        GER = {
            or = { has_war_with = FRA has_war_with = ENG }
            or = { is_ally_with = AUS is_ally_with = TUR }
        }
    }
    fire_only_once = yes

    option = { news_event = _war.1 }
}

news_event = {
    id = _war.1
    title = "Great War in Europe!"
    desc = _war.1.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "The old world is engulfed in fire."
        if = {
            limit = { or = { tag = GER is_ally_with = GER has_war_with = GER } }
            add_timed_idea = { idea = idea_war_enthusiasm months = 6 }

            hidden_effect = {
                random = { chance = 25 news_event = { id = _war.2 days = 7 random_days = 7 } }
                random = { chance = 25 news_event = { id = _war.3 days = 14 random_days = 7 } }
            }
        }
    }
}

## -- armistice
add_namespace = _armistice

news_event = {
    id = _armistice.0
    hidden = yes 

    is_triggered_only = yes
    trigger = {
        surrender_progress < 0.05
        not = { any_enemy_country = { surrender_progress < 0.05 } }
    }

    option = { country_event = _armistice.1 }
}

country_event = {
    id = _armistice.1
    title = "Time to Seek an Armistice?"
    desc = _armistice.1.d

    is_triggered_only = yes
    trigger = { has_war = yes has_civil_war = no }

    option = {
        name = "The time has come."
        every_enemy_country = {
            limit = { or = { is_faction_leader = yes is_in_faction = no } }
            country_event = { id = _armistice.2 hours = 6 random_days = 7 } 
        }
        ai_chance = { 
            factor = 2
            modifier = { has_war_support > 0.5 factor = 0.5 }
        }
    }
    option = {
        name = "We shall continue fighting."
        add_political_power = -75

        ai_chance = { 
            factor = 1
            modifier = { 
                and = { original_tag = FRA not = { controls_state = 16 } }      # FRA surrenders
                factor = 0
            }
        }
    }
}

country_event = {
    id = _armistice.2
    title = "[FROM.GetName] Offers an Armistice"
    desc = _armistice.2.d

    is_triggered_only = yes
 
    option = {
        name = "We agree."
        start_peace_conference = { tag = FROM score_factor = 0.2 }
        FROM = { leave_faction = yes }
        
        ai_chance = { factor = 4 }
    } 
    option = {
        name = "Ignore them."
        add_political_power = -75
        
        ai_chance = { 
            factor = 1
            modifier = { FROM = { is_faction_leader = yes } factor = 0 }        # AI will always accept factions surrendering.
        }
    }
}
