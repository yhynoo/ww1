add_namespace = germany_winter

country_event = {
    id = germany_winter.1
    title = "Winter of 1916"
    desc = germany_winter.1.d

    trigger = {
        tag = GER
        and = { 
            has_war = yes
            not = { has_idea = civilian_economy }
        }
        and = { 
            date > 1916.9.1 
            date < 1917.1.1 
        }
    }
    fire_only_once = yes

    option = {
        name = "Demobilize the economy."

        add_ideas = civilian_economy
        add_popularity = { ideology = democratic popularity = 0.05 }
    }
    option = {
        name = "We must endure."
        
        add_political_power = 75
        add_timed_idea = { idea = idea_winter_of_1916 months = 6 }

        hidden_effect = {
            news_event = { id = germany_winter.2 days = 60 random_days = 14 }
            country_event = { id = germany_winter.3 days = 180 random_days = 14 }
        }
    }
}

news_event = {
    id = germany_winter.2
    title = "Hunger in Hamburg"
    desc = germany_winter.2.d

    is_triggered_only = yes

    option = {
        name = "We must all make sacrifices."
        add_stability = -0.025
    }
}

country_event = {
    id = germany_winter.3
    title = "After the Winter"
    desc = germany_winter.3.d

    is_triggered_only = yes

    option = {
        name = "Maintain the course."

        add_stability = -0.05
        add_popularity = { ideology = communism popularity = 0.05 }
    }
    option = {
        trigger = {
            or = {
                has_government = monarchism
                and = { authoritarianism < 0.5 democratic < 0.5 communism < 0.5 }
            }
        }
        name = "Appoint Ludendorff."
        promote_character = GER_erich_ludendorff
    }
    option = {
        trigger = {
            not = { has_government = monarchism }
        }

        name = "Seek an armistice."

        every_enemy_country = {
            limit = {
                or = { 
                    is_faction_leader = yes 
                    not = { any_country = { has_war_together_with = FROM } }
                } 
            }
            country_event = { id = _offer_faction_leader.1 hours = 6 random_hours = 6 }
        }
    }
}
