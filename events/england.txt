### 1. Irish Home Rule

news_event = {
    id = england_events.1
    title = "United Kingdom Grants Ireland Independence"
    desc = england_events.1.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "A new era for Ireland."
    }
}

news_event = {
    id = england_events.2
    title = "United Kingdom Suppresses Irish Autonomy"
    desc = england_events.2.d

    is_triggered_only = yes

    option = {
        name = "A firm stance."
        hidden_effect = {
            country_event = { id = england_events.3 days = 7 random_days = 7 }
        }
    }
}

country_event = {
    id = england_events.3
    title = "Riots in Cork"
    desc = england_events.3.d

    is_triggered_only = yes

    trigger = {
        has_stability < 0.8
    }

    option = {
        name = "Use force."
        add_stability = -0.05

        hidden_effect = {
            load_oob = "ENG_irish_unionists"
            release = IRE

            119 = { add_claim_by = IRE }
            add_named_threat = {
                name = "Independence war in Ireland"
                threat = 3
            }

            declare_war_on = {
                target = IRE
                type = annex_everything
            }
            
            IRE = { load_oob = "IRE_1912" }
        }

        ai_chance = { base = 2 }
    }

    option = {
        add_political_power = -150
        name = "Grant more autonomy."

        ai_chance = { 
            base = 1
            modifier = {
                or = {
                    communism > 0.3
                    has_war_support < 0.4
                }
                factor = 5
            }
        }
    }
}

        news_event = { id = england_events.6 hours = 12 random_hours = 6 }

        ai_chance = {
            base = 1
            modifier = {
                ENG = { has_country_flag = announced_containment }

            }
            modifier = {

            }

                tag = JAP

            }
        }

        name = "No, we must reconsider."

                target = ENG

            }
        }

        ai_chance = { 
            base = 1
            modifier = {
                ENG = { has_war = yes }
                factor = 3
            }
        }
    }
}
