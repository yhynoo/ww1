country_event = {
    id = russia_events.2
    title = "[SOV.GetName] Offers Alliance!"
    desc = russia_events.2.d

    is_triggered_only = yes

    option = {
        name = "They will guarantee our future!"
        SOV = {
            if = {
                limit = { is_in_faction = no }
                create_faction = "Belgrade Pact"
            }
            faction_leader = { add_to_faction = SER }
        }

        # generate the announcement in world news:
        news_event = { id = russia_events.3 hours = 12 random_hours = 6 }

        ai_chance = { 
            base = 3
            modifier = {
                OR = {
                    has_opinion = {
                        target = AUS
                        value < 0
                    }
                    has_opinion = {
                        target = TUR
                        value < 0
                    }
                }
                factor = 3
            }
        }
    }

    option = {
        name = "This may lead to a great war..."

        # generate response event in russia:
        SOV = { news_event = { id = russia_events.4 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 1
            modifier = {
                has_opinion = {
                    target = AUS
                    value > 50
                }
                factor = 5
            }
        }
    }
}

news_event = {
    id = russia_events.5
    title = "Infiltration of Manchuria"
    desc = russia_events.5.d

    trigger = {
        not = { has_war_with = MAN }
        any_state = {
            or = {
                state = 328
                state = 714
                state = 716
                state = 717
                state = 761
            }
            is_owned_by = MAN
        }
    }

    is_triggered_only = yes
    major = yes

    option = {
        name = "Interesting."

        if = {
            limit = { tag = SOV }
            hidden_effect = {
                add_named_threat = {
                    threat = 1
                    name = "Infiltration of Manchuria"
                }
            }
        }
    }
}

country_event = {
    id = russia_events.6
    title = "Manchuria or War!"
    desc = russia_events.6.d

    is_triggered_only = yes

    immediate = {
        SOV = {
            add_named_threat = {
                threat = 3
                name = "Territorial Dispute over Manchuria"
            }
        }
    }

    option = {
        name = "We must concede."
        news_event = { id = russia_events.7 hours = 12 random_hours = 6 }

        ## here goes the concession
        every_state = {
            limit = {
                OR = {
                    state = 328
                    state = 714
                    state = 716
                    state = 717
                    state = 761
                }
                is_owned_by = MAN
            }
            transfer_state_to = SOV
        }
        
        ai_chance = { 
            base = 1
            modifier = {
                strength_ratio = {
                    tag = SOV
                    ratio < 1
                }
                factor = 2.5
            }
            modifier = {
                strength_ratio = {
                    tag = SOV
                    ratio < 0.5
                }
                factor = 5
            }
            modifier = {
                has_war = yes           # if they're at war, they will just cede and forget about it.
                factor = 100
            }
        }
    }

    option = {
        name = "Prepare for war!"
        news_event = { id = russia_events.8 hours = 12 random_hours = 6 }

        SOV = { 
            create_wargoal = {
                target = MAN
                type = take_claimed_state
            }
        }

        ai_chance = { 
            base = 1
            modifier = {
                strength_ratio = {
                    tag = SOV
                    ratio > 0.5
                }
                factor = 2.5
            }
            modifier = {
                strength_ratio = {
                    tag = SOV
                    ratio > 1
                }
                factor = 5
            }
            modifier = {
                is_in_faction = yes
                factor = 2
            }
        }
    }
}

country_event = {
    id = russia_events.9
    title = "[SOV.GetNameDef] Demands Mongolian Independence"
    desc = russia_events.9.d

    is_triggered_only = yes

    immediate = {
        SOV = {
            add_named_threat = {
                threat = 3
                name = "Demands for Mongolian Independence"
            }
        }
    }

    option = {
        name = "Yes, Mongolia shall be free."

        # create Mongolia
        every_state = {
            limit = { is_core_of = MON }
            transfer_state_to = MON
        }
        SOV = { puppet = MON }

        # and make announcement
        news_event = { id = russia_events.10 hours = 12 random_hours = 6 }

        ai_chance = {
            base = 2
            modifier = {
                has_stability < 0.4
                factor = 2.5
            }
            modifier = {
                has_war = yes       # if they're at war, no time to deal with this stuff
                factor = 100 
            }
        }
    }

    option = {
        trigger = {
            is_in_faction = no
            SOV = {
                OR = {
                    is_faction_leader = yes
                    is_in_faction = no
                }
            }
        }

        name = "Let us propose an alliance."
        SOV = { country_event = { id = russia_events.11 hours = 12 random_hours = 6 } }

        ai_chance = {
            base = 2
            modifier = {
                has_opinion = {
                    target = SOV
                    value > 25
                }
                factor = 4
            }
            modifier = {
                SOV = { is_in_faction = yes }
                factor = 2
            }
        }
    }

    option = {
        name = "Absolutely not!"
        SOV = { news_event = { id = russia_events.14 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 4

            limit = { is_in_faction = no }

        }
        add_to_faction = MAN

        # add the world news
        news_event = { id = russia_events.12 hours = 12 random_hours = 6 }

        ai_chance = {
            base = 2
            modifier = {
                date > 1916.1.1
                factor = 2
            }
            modifier = {
                strength_ratio = {
                    tag = MAN        # if China is strong enough - why not?

                }
                factor = 4
            }
        }
    }

    option = {
        name = "This is not what we wanted."
        MAN = { news_event = { id = russia_events.13 hours = 12 random_hours = 6 } }  # this is news only in China.

            base = 1
            modifier = {
                strength_ratio = {
                    tag = MAN        # but if not, then we'd rather fight them
                    ratio > 2
                }
                factor = 8
            }
        }
    }
}
