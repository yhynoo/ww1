add_namespace = manchukuo_warlords

news_event = {
    id = manchukuo_warlords.1
    title = "The Anhui Clique Declares Independence!"
    desc = manchukuo_warlords.1.d

    trigger = {
        and = {
            date > 1917.7.1
            not = { has_country_leader = { character = MAN_duan_qirui ruling_only = yes } }
        }
    }
    fire_only_once = yes

    option = {
        name = "To arms!"

        hidden_effect = { 
            anhui_independence = yes 
            retire_character = MAN_duan_qirui
        }

        declare_war_on = { target = ANH type = annex_everything }
    }
}
