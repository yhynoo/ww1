add_namespace = spain_elections

country_event = {
    id = spain_elections.1
    title = "The Election of 1914"
    desc = spain_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = SPA
        has_government = democratic

        date > 1914.1.1         date < 1915.1.1
    }

    option = {
        name = "Maintain ‘turnismo.’"
        add_stability = 0.05

        promote_character = SPA_eduardo_dato

        ai_chance = { base = 1 }
    }
    option = {
        name = "Álvaro de Figueroa, the Count of Romanones."

        add_popularity = { ideology = democratic popularity = 0.05 }
        promote_character = SPA_alvaro_de_figueroa

        ai_chance = { base = 0 }
    }
}

country_event = {
    id = spain_elections.2
    title = "The Election of 1918"
    desc = spain_elections.2.d

    is_triggered_only = yes
    trigger = {
        tag = SPA
        has_government = democratic

        date > 1918.1.1         date < 1919.1.1
    }

    option = {
        name = "Maintain ‘turnismo.’"
        add_stability = 0.05

        if = {
            limit = { has_country_leader = { character = SPA_eduardo_dato ruling_only = yes } }
            promote_character = SPA_alvaro_de_figueroa
        }
        else = {
            promote_character = SPA_antonio_maura
        }

        ai_chance = { base = 2 }
    }
    option = {
        trigger = { has_country_leader = { character = SPA_eduardo_dato ruling_only = yes } }
        name = "Antonio Maura."
        
        add_popularity = { ideology = monarchism popularity = 0.05 }
        promote_character = SPA_antonio_maura
    }
    option = {
        trigger = { has_country_leader = { character = SPA_alvaro_de_figueroa ruling_only = yes } }
        name = "Álvaro de Figueroa, the Count of Romanones."
        
        add_popularity = { ideology = democratic popularity = 0.05 }
        promote_character = SPA_alvaro_de_figueroa
    }
}
