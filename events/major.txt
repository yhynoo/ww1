add_namespace = majorevent

news_event = {
	id = majorevent.1
	title = "Arson in Brussels: Political Motives?"
	desc = majorevent.1.d

    trigger = {
        HOL = { owns_state = 6 }
    }

	major = yes
	is_triggered_only = yes

	option = {
		if = {
			limit = { tag = FRA }
			add_opinion_modifier = {
				target = GER
				modifier = medium_decrease
			}
		}
		if = {
			limit = { tag = HOL }
			add_stability = -0.05
		}
		if = {
			limit = { tag = GER }
			add_opinion_modifier = {
				target = FRA
				modifier = small_decrease
			}

			hidden_effect = {
				add_named_threat = {
					threat = 5
					name = "Arson of the French Embassy in Brussels"
				}
			}
		}

		name = "A troubling incident."
	}
}

news_event = {
	id = majorevent.2
	title = "Franz Ferdinand Assassinated!"
	desc = majorevent.2.d

	major = yes
	is_triggered_only = yes

	trigger = {
		AUS = { has_government = monarchism }
	}

	option = {
		if = {
			limit = { tag = AUS }
			add_opinion_modifier = {
					target = SOV
					modifier = small_decrease
				}

			if = {
				limit = { SER = { exists = yes } }
				add_opinion_modifier = {
					target = SER
					modifier = medium_decrease
				}
			}
			if = {
				limit = { has_idea = sww1_idea_slavic_unrest }
				create_wargoal = {
					target = SOV
					type = annex_everything
				}
			}

			hidden_effect = {
				add_named_threat = {
					threat = 12
					name = "Assassination of Franz Ferdinand"
				}
			}
		}
		name = "A shocking act of violence!"
	}
}

news_event = {
	id = majorevent.3

	desc = majorevent.3.d

	major = yes
	is_triggered_only = yes

	trigger = {
		AUS = { has_government = monarchism }
	}

	option = {
        trigger = {
            not = { tag = AUS }
        }
		name = "A new era for Austria-Hungary."
	}
    option = {
        trigger = {
            tag = AUS
        }
        name = "Long live the Emperor!"
        retire_character = AUS_franz_joseph
        promote_character = AUS_karl_von_habsburg
    }
}
