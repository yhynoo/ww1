add_namespace = _demand

country_event = {
    id = _demand.1
    title = "[FROM.GetName] Demands Territory"
    desc = _demand.1.d

    is_triggered_only = yes

    option = {
        name = "We must accept their demands."
        FROM = { news_event = { id = _demand.2 hours = 6 random_hours = 6 } }

        every_controlled_state = {
            limit = { or = { is_claimed_by = FROM is_core_of = FROM } }
            transfer_state_to = FROM
        }

        ai_chance = {
            base = 1

            # we will be more likely to give in if they are much stronger than us
            modifier = { strength_ratio = { tag = FROM ratio > 2 } factor = 2 }

            # Germany and USA won't.
            modifier = { 
                or = { 
                    and = { tag = GER not = { has_government = democratic } }
                    tag = USA
                } 
                factor = 0
            }

            # Holland won't, if they decided to go monarchist.
            modifier = { has_completed_focus = HOL_adjust_the_government factor = 0 }
        }
    }

    option = {
        name = "Ignore them."
        add_war_support = 0.05
        
        FROM = { create_wargoal = { target = ROOT type = take_claimed_state } }

        ai_chance = {
            base = 2

            # if they are much stronger, we might rethink
            modifier = { strength_ratio = { tag = FROM ratio < 0.5 } factor = 0.5 }

            modifier = { 
                or = {
                    has_civil_war = yes
                    and = { tag = HOL not = { has_completed_focus = HOL_adjust_the_government } }       # Holland hasn't gone monarchist.
                }
                factor = 0 
            }
        }

        hidden_effect = { 
            FROM = { add_named_threat = { threat = 3 name = "[ROOT.GetName] Ignores [FROM.GetAdjective] Demands" } }
        }
    }
}

news_event = {
    id = _demand.2
    title = "[FROM.GetName] Yields Land!"
    desc = _demand.2.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "A victory for [ROOT.GetNameDef]."
    }
}
