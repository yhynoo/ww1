## Alliance with France

country_event = {
    id = austria_second_vienna_treaty.3
    title = "Austria-Hungary Offers an Alliance"
    desc = austria_second_vienna_treaty.3.d

    is_triggered_only = yes

    option = {
        name = "Accept the alliance offer."
        AUS = { 
            news_event = { id = austria_second_vienna_treaty.4 hours = 12 random_hours = 6 }

            if = {
                limit = { is_in_faction = no }
                create_faction = "Second Vienna Treaty"
            }
            add_to_faction = FRA
        }

        ai_chance = { base = 6 }

        hidden_effect = {
            news_event = { id = austria_second_vienna_treaty.9 days = 14 random_days = 7 }
        }
    }

    option = {
        name = "Reject the alliance offer."
        AUS = { news_event = { id = austria_second_vienna_treaty.5 } hours = 12 random_hours = 6 }

        ai_chance = { base = 1 }
    }
}
