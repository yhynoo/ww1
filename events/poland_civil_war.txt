add_namespace = poland_civil_war

news_event = {
    id = poland_civil_war.1
    title = "Anti-Russian Uprising in Poland!"            # proclamation of the state
    desc = poland_civil_war.1.d

    trigger = { POL = { exists = no } }

    is_triggered_only = yes
    major = yes

    option = {
        name = "Poland is not yet lost!"

        if = { 
            limit = { original_tag = SOV }
            hidden_effect = {
                every_owned_state = {
                    limit = {
                        or = { state = 10 state = 87 state = 97 state = 814 }
                    }
                    transfer_state_to = POL
                }

                declare_war_on = {
                    target = POL
                    type = annex_everything
                }
            }
        }

        if = {
            limit = { tag = SOV }
            hidden_effect = { 
                POL = { 
                    news_event = { id = poland_civil_war.2 hours = 6 }

                    country_event = { id = poland_civil_war.3 days = 7 random_hours = 12 }
                    country_event = { id = poland_civil_war.4 days = 14 random_days = 7 }
                }
            }
        }
    }
}

# Request return of POWs

country_event = {
    id = poland_civil_war.4
    title = "Request Return of POWs?"
    desc = poland_civil_war.4.d

    is_triggered_only = yes 
    trigger = {
        any_country = {
            or = { tag = AUS tag = GER }
            not = { has_war_with = POL }
        }
    }

    option = {
        name = "Let us ask."

        add_political_power = -100
        every_country = {
            limit = {
                or = { tag = AUS tag = GER }
                not = { has_war_with = POL }
            }
            country_event = { id = poland_civil_war.5 hours = 12 random_hours = 6 }
        }

        ai_chance = { base = 6 }
    }

    option = {
        name = "This is not important now."
        ai_chance = { base = 1 }
    }
}
