    id = germany_atlantic.1
    title = "Navy to Expand Submarine Warfare?"
    desc = germany_atlantic.1.d

    trigger = {
        tag = GER
        war_length_with = { tag = ENG months > 6 }
    }
    fire_only_once = yes

    option = {
        name = "Gott Strafe England!"
        swap_ideas = {
            remove_idea = idea_tirpitz_plan
            add_idea = idea_atlantic_submarine_warfare
        }

        hidden_effect = { 
            random = { 
                chance = 80 
                USA = { news_event = { id = germany_atlantic.2 days = 60 random_days = 60 } }
            }

            random = { 
                chance = 20
                SPA = { news_event = { id = germany_atlantic.3 days = 60 random_days = 60 } }
            }

            random = { 
                chance = 20
                ARG = { news_event = { id = germany_atlantic.3 days = 60 random_days = 60 } }
            }
        }
        ai_chance = { base = 3 }
    }
    option = {
        name = "This is too risky."
        add_political_power = -25

        ai_chance = { base = 1 }
    }
}
