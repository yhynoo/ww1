# Netherlands

country_event = {
    id = france_belgian_crisis.1
    title = "France Demands We Leave Flanders"
    desc = france_belgian_crisis.1.d

    is_triggered_only = yes

    option = {
        name = "This is a welcome development."
        FROM = { 
            news_event = { id = france_belgian_crisis.2 hours = 6 random_hours = 6 }
            every_state = {
                limit = { or = { state = 6 state = 34 } is_owned_and_controlled_by = HOL }      # add states.
                transfer_state_to = FRA
            }
        }

        ai_chance = { base = 1 modifier = { has_completed_focus = HOL_thorbeckes_legacy factor = 3 } }
    }
    option = {
        name = "We will not agree to this."
        FROM = { 
            news_event = { id = france_belgian_crisis.3 hours = 6 random_hours = 6 }
            create_wargoal = {
                target = ROOT
                type = take_state_focus
            }
        }

        ai_chance = { base = 1 modifier = { has_completed_focus = HOL_integrate_the_belgians factor = 3 } }
    }
}
