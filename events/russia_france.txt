add_namespace = russia_france

country_event = {
    id = russia_france.1
    title = "[FROM.GetName] Seeks Military Alliance"
    desc = russia_france.1.d 

    is_triggered_only = yes

    option = {
        name = "That's a good idea."
        FROM = { news_event = { id = russia_france.2 hours = 12 random_hours = 6 } }
        
        if = {
            limit = { is_in_faction = no }
            create_faction = "Versailles Accords"
        }
        add_to_faction = FROM

        hidden_effect = {
            FROM = {
                add_ai_strategy = {
                    type = alliance
                    id = ROOT
                    value = 200
                }
            }
        }

        ai_chance = { 
            base = 6
            modifier = { has_opinion = { target = FROM value > 50 } factor = 2 }
            modifier = { has_war = no factor = 0 }
        }
    }

    option = {
        name = "We are only interested in mutual defensive guarantees."
        trigger = {
            NOT = { 
                OR = { has_guaranteed = FROM FROM = { has_guaranteed = ROOT } }
            }
        }

        FROM = { country_event = { id = russia_france.3 hours = 12 random_hours = 6 } }
        ai_chance = { base = 1 }
    }

    option = {
        name = "We are not interested."
        FROM = { news_event = { id = russia_france.6 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 1
            modifier = { has_opinion = { target = FROM value < 0 } factor = 3 }
        }
    }
}
