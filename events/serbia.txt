# join the war?
add_namespace = ser_war

country_event = {
    id = ser_war.1
    title = "Opportunity to Strike Austria?"
    desc = ser_war.1.d

    trigger = {
        tag = SER
        not = { has_country_flag = join_event_done }
        and = { 
            has_war = no 
            is_subject = no 
            AUS = { 
                has_war_with = SOV 
                war_length_with = { tag = SOV months < 2 } 
            }
        }
    }
    immediate = { set_country_flag = join_event_done }

    option = {
        name = "This is the moment!"
        declare_war_on = { target = AUS type = take_claimed_state }
        add_timed_idea = { idea = idea_war_enthusiasm months = 6 }

        hidden_effect = { 
            news_event = { id = ser_war.2 days = 7 random_days = 7 }
        }
    }
    option = {
        name = "Insane."
        add_popularity = { ideology = democratic popularity = 0.025 }
    }
}

news_event = {
    id = ser_war.2
    title = "King in the Trenches"
    desc = ser_war.2.d

    is_triggered_only = yes
    trigger = { has_government = monarchism }

    option = {
        name = "A true Serbian soul."
        add_popularity = { ideology = monarchism popularity = 0.025 }
        add_war_support = 0.05
    }
}
