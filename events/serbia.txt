# - agrarianists
add_namespace = ser_agrarianists

country_event = {
    id = ser_agrarianists.1
    title = "Agrarianists Demand Investments"
    desc = ser_agrarianists.1.d

    is_triggered_only = yes

    option = {
        name = "This is a good idea."
        add_popularity = { ideology = communism popularity = 0.05 }
        
        random_owned_controlled_state = {
            add_building_construction = { type = infrastructure level = 1 instant_build = 1 }
            add_resource = { type = tungsten amount = 4 }
        }
        ai_choice = { factor = 1 }
    }
    option = {
        name = "We need our attention elsewhere."
        add_popularity = { ideology = communism popularity = -0.025 }
        ai_choice = { factor = 0 }
    }
}

# - drina: serbian intelligence is trying too hard maybe?
add_namespace = ser_drina

country_event = {
    id = ser_drina.1
    title = "Activity Across the Drina"
    desc = ser_drina.1.d

    is_triggered_only = yes
    trigger = { 
        not = { has_war_with = AUS }
        AUS = { controls_state = 103 } 
    }

    option = {
        name = "This is too much."
        add_war_support = -0.025   
    }
    option = {
        name = "Very well."
        add_timed_idea = { idea = idea_activity_across_drina years = 1 }

        hidden_effect = { 
            random = { chance = 50 news_event = { id = ser_drina.2 days = 7 random_days = 7 } }
        }
    }
}

news_event = {
    id = ser_drina.2
    title = "Serbian Agents Discovered in Mostar!"
    desc = ser_drina.2.d

    is_triggered_only = yes
    major = yes
    
    trigger = { AUS = { controls_state = 103 } }

    option = {
        name = "Very worrying."
        if = {
            limit = { tag = AUS }
            create_wargoal = { target = SER type = annex_everything }
        }
        if = {
            limit = { tag = SER }
            add_named_threat = { name = "Serbian Agents Discovered in Mostar!" threat = 6 }
        }
    }
}
