add_namespace = _dutch_neutrality

news_event = {
    id = _dutch_neutrality.1
    title = "Debates About Neutrality"
    desc = _dutch_neutrality.1.d

    trigger = {
        tag = HOL
        is_in_faction = no

        not = { has_war_with = ENG }
        not = { has_war_with = GER }

        GER = { has_war_with = ENG }
    }
    fire_only_once = yes

    option = {
        name = "Alright."
        set_power_balance = { 
            id = HOL_dutch_neutrality 
            set_value = -0.1
        }
    }
}

country_event = {
    id = _dutch_neutrality.2
    title = "Dutch Trade Fleet"
    desc = _dutch_neutrality.2.d

    trigger = {
        tag = ENG
        HOL = { 
            power_balance_value = { id = HOL_dutch_neutrality value < -0.25 }       # HOL is slightly pro-GER
            has_equipment = { convoy > 20 }
        }
    }
    fire_only_once = yes
    mean_time_to_happen = { months = 3 }

    option = {
        name = "We shall do it."
        add_equipment_to_stockpile = { type = convoy    amount = 20     producer = HOL }

        HOL = { news_event = { id = _dutch_neutrality.3 hours = 6 } }
    }
    option = {
        name = "This is not needed."
    }
}

# East Indies

country_event = {
    id = _dutch_neutrality.4
    title = "Demand Dutch East Indies?"
    desc = _dutch_neutrality.4.d

    trigger = {
        tag = ENG
        HOL = { 
            power_balance_value = { id = HOL_dutch_neutrality value < -0.65 }   # HOL is very pro-GER
            any_owned_state = { is_on_continent = asia }
        }
    }
    fire_only_once = yes
    mean_time_to_happen = { months = 3 }

    option = {
        name = "We shall do it."

        every_state = {
            limit = { 
                is_on_continent = asia
                is_owned_by = HOL
            }
        }

        HOL = { news_event = { id = _demand.1 hours = 6 } }
    }
    option = {
        name = "This is not needed."
    }
}
