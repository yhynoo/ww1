### 1. Japan wants to Entente

country_event = {
    id = japan_events.1
    title = "Japan Seeks to Join the Entente"
    desc = japan_events.1.d

    is_triggered_only = yes

    option = {
        name = "They are welcome."
        add_to_faction = JAP

        news_event = { id = japan_events.2 hours = 12 random_hours = 6 }
        ai_chance = {
            base = 3
            modifier = {
                threat > 0.1
                factor = 2
            }
            modifier = {
                has_war = yes
                factor = 3
            }
        }
    }

    option = {
        name = "This is not the right time."

        news_event = { id = japan_events.2 hours = 12 random_hours = 6 }
        ai_chance = { base = 1 }
    }
}

### 2. Twenty-One Demands

country_event = {
    id = japan_events.7
    title = "Japan Demands Payment in Resources for Debt"
    desc = japan_events.7.d

    is_triggered_only = yes

    option = {
        name = "Agree to the settlement."
        news_event = { id = japan_events.8 hours = 12 random_hours = 6 }

        give_resource_rights = { receiver = JAP state = 592 }
        give_resource_rights = { receiver = JAP state = 593 }

        ai_chance = {
            base = 3
            modifier = {
                is_in_faction = yes             # if we got backup, we will fight.
                factor = 0
            }
        }
    }

    option = {
        name = "We will fight."
        JAP = { news_event = { id = japan_events.9 hours = 12 random_hours = 6 } }

        JAP = {
            create_wargoal = {
                target = MAN
                type = annex_everything
            }
        }

        hidden_effect = {
            add_named_threat = {
                name = "Tensions Grow in East Asia!"
                threat = 5
            }
        }

        ai_chance = { base = 1 }
    }
}
