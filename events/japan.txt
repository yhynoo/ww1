            base = 1

                has_war = yes

            }
        }

        name = "No, refuse their demand."
        
        JAP = {
            news_event = { id = japan_events.3 hours = 12 random_hours = 6 }
            create_wargoal = {
                target = GER

            }
        }

        ai_chance = {
            base = 2
            modifier = {
                or = {
                    is_ally_with = SOV
                    is_in_faction_with = MAN
                }
                factor = 20
            }
        }
    }
}
