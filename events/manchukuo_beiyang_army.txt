country_event = {
    id = manchukuo_beiyang_army.3
    title = "Uprising in Mongolia, Monarchists Threaten War!"
    desc = manchukuo_beiyang_army.3.d

    trigger = {
        tag = MAN
        has_country_flag = ready_for_civil_war
        monarchism > 0.4
    }

    immediate = {
        hidden_effect = {
            every_state = {
                limit = {
                    or = {
                        state = 330
                        state = 819
                        state = 820
                    }
                    is_owned_by = MAN
                }
                transfer_state_to = MON
            }

            MON = { load_oob = "MON_uprising" }
            declare_war_on = {
                target = MON
                type = annex_everything
            }
        }
    }

    option = {
        name = "We must acknowledge our failure."
        set_politics = {
            ruling_party = monarchism
            elections_allowed = no
        }

        hidden_effect = { clr_country_flag = ready_for_civil_war }

        ai_choice = { base = 2 }
    }
    option = {
        name = "We must defend our democracy!"
        start_civil_war = {
            ideology = monarchism
            size = 0.5
        }

        hidden_effect = { 
            add_popularity = {
                ideology = monarchism
                popularity = -0.4
            }
            random_country = {
                limit = { original_tag = MAN has_government = monarchism }
                set_cosmetic_tag = MAM

                declare_war_on = {
                    target = MON
                    type = annex_everything
                }
            }
            
            clr_country_flag = ready_for_civil_war
        }

        ai_choice = { base = 1 }
    }
}
