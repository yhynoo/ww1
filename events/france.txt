# Ask the Dutch to Leave
country_event = {
    id = france_events.1
    title = "[FRA.GetName] Asks the Dutch to Leave"
    desc = france_events.1.d

    is_triggered_only = yes

    option = {
        name = "Let them take over in Belgium."

        news_event = { id = france_events.2 hours = 12 random_hours = 6 }
        every_state = {
            limit = {
                or = {
                    state = 6
                    state = 34
                    state = 977
                    state = 980
                }
                is_owned_by = HOL
            }
            transfer_state_to = FRA
        }
        remove_ideas = sww1_idea_belgian_crisis

        ai_chance = {
            base = 5
            modifier = {
                is_ally_with = GER          # won't give up if they're allied with Germany
                factor = 0
            }
        }
    }

    option = {
        name = "We refuse to yield!"
        FRA = {
            news_event = { id = france_events.3 hours = 12 random_hours = 6 }
        }

        hidden_effect = {
            add_named_threat = {
                threat = 2
                name = "The Dutch Won't Leave Belgium"
            }
        }

        ai_chance = {
            base = 1
        }
    }
}

    id = france_events.17
    title = "[FRA.GetName] Proposes Unification with [SPR.GetName]"
    desc = france_events.17.d

    is_triggered_only = yes

    option = {
        name = "We shall unite with France!"

        FRA = {
            add_political_power = 100
            annex_country = {
                target = SPR
                transfer_troops = yes
            }
        }

        hidden_effect = {
            ENG = {
                country_event = { id = france_events.20 days = 7 random_hours = 6 }
            }
            USA = {
                country_event = { id = france_events.20 days = 7 random_hours = 6 }
            }

            add_named_threat = {
                threat = 3
                name = "Franco-Spanish Unification"
            }
        }

        ai_chance = {
            base = 1
            modifier = {
                has_opinion = {
                    target = FRA
                    value > 50
                }
                factor = 2
            }
            modifier = {
                FRA = { is_ally_with = AUS }
                factor = 5
            }
        }
    }

    option = {
        name = "We must remain independent."
        FRA = {
            news_event = { id = france_events.19 hours = 12 random_hours = 6 }
            add_opinion_modifier = {
                target = SPR
                type = medium_decrease
            }
        }

        ai_chance = {
            base = 2
        }
    }
}
