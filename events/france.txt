## BELGIAN CRISIS

country_event = {
	id = france_events.1
	title = "France Demands We Leave Belgium"
	desc = france_events.1.d

	is_triggered_only = yes

	option = {
		name = "We shall withdraw."
		news_event = { id = france_events.2 hours = 12 random_hours = 6 }

        every_state = {
            limit = {
                OR = {
                    state = 6
                    state = 34
                    state = 977
                    state = 980
                }
                is_owned_by = GER
            }
            transfer_state_to = FRA
        }

        ai_chance = { 
            base = 1
            modifier = {
                strength_ratio = {
                    tag = FRA
                    ratio > 1
                }
                factor = 10
            }
            modifier = {
                FRA = { 
                    OR = {
                        is_ally_with = SOV
                        is_ally_with = AUS
                    } 
                }
                factor = 3
            }
        }
	}

	option = {
		name = "No, we will hold our ground."
		FRA = { country_event = { id = france_events.3 hours = 12 random_hours = 6 } }

        ai_chance = { base = 5 }
	}
}

country_event = {
	id = france_events.3
	title = "Germans Decline!"
	desc = france_events.3.d

	is_triggered_only = yes

    immediate = {
        add_named_threat = {
            threat = 3
            name = "[GER.GetName] Stays in the Ardennes"
        }
    }

	option = {
		name = "This means war!"
		news_event = { id = france_events.4 hours = 12 random_hours = 6 }

        declare_war_on = {
            target = GER
            type = take_claimed_state
        }

        ai_chance = {
            base = 1
            modifier = {
                OR = {
                    is_ally_with = SOV
                    is_ally_with = AUS
                }
                factor = 3
            }
        }

        hidden_effect = {
            add_named_threat = {
                threat = 25
                name = "[FRA.GetName] Goes to War!"
            }
        }
	}

	option = {
        add_political_power = -50
		name = "We must back down."

		hidden_effect = {
            GER = { country_event = { id = france_events.5 days = 7 random_days = 14 } }
        }

        ai_chance = { base = 3 }
	}
}
