## -- agadir
add_namespace = fra_agadir

country_event = {
    id = fra_agadir.1
    title = "Caillaux’ Dismissal"
    desc = fra_agadir.1.d

    trigger = { tag = FRA   date > 1912.2.1 }
    fire_only_once = yes

    option = {
        name = "Dismiss him."
        promote_character = FRA_aristide_briand
    }
    option = {
        name = "He deserves another chance."
        add_popularity = { ideology = democratic popularity = -0.02 }

        hidden_effect = { news_event = { id = fra_morocco.1 days = 14 random_days = 7 } }
    }
}

### ELECTIONS ###
add_namespace = fra_elections

country_event = {
    id = fra_elections.1
    title = "The Election of 1913"
    desc = fra_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = FRA
        has_government = democratic
        date > 1913.1.1         date < 1914.1.1
    }

    option = {
        name = "Louis Barthou."
        promote_character = FRA_louis_barthou
        ai_chance = { factor = 1 }
    }
    option = {
        name = "Aristide Briand."
        if = {
            limit = { has_country_leader = { character = FRA_aristide_briand ruling_only = yes } }
            add_stability = 0.05
        }
        else = { promote_character = FRA_aristide_briand }
        ai_chance = { factor = 2 }
    }
}
