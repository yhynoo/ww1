# - morocco
add_namespace = fra_morocco

news_event = {
    id = fra_morocco.1
    title = "Arson in Agadir!"
    desc = fra_morocco.1.d

    trigger = {
        tag = FRA
        date > 1912.4.1
        and = { has_idea = idea_unrest_in_morocco has_stability < 0.5 not = { has_country_flag = arson_event_done } }
    }
    immediate = { set_country_flag = arson_event_done }

    option = {
        name = "Worrying."
        add_stability = -0.025
        hidden_effect = { random = { chance = 80 country_event = { id = fra_morocco.2 days = 7 random_days = 7 } } }
    }
}

country_event = {
    id = fra_morocco.2
    title = "Morocco Demands Independence!"
    desc = fra_morocco.2.d

    is_triggered_only = yes

    option = {
        name = "To arms!"

        hidden_effect = { 
            moroccan_uprising = yes
            SPA = { country_event = { id = spa_morocco.1 days = 7 } }
        }
        declare_war_on = { target = MOR type = annex_everything }
    }
    option = {
        name = "We cannot afford a war over Morocco."
        release = MOR
    }
}

    id = fra_elections.1
    title = "The Election of 1913"
    desc = fra_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = FRA
        has_government = democratic
        date > 1913.1.1         date < 1914.1.1
    }

    option = {
        name = "Louis Barthou."
        promote_character = FRA_louis_barthou
        ai_chance = { factor = 1 }
    }
    option = {
        name = "Aristide Briand."

            limit = { has_country_leader = { character = FRA_aristide_briand ruling_only = yes } }
            add_stability = 0.05

        else = { promote_character = FRA_aristide_briand }
        ai_chance = { factor = 2 }
    }
}
