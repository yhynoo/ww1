        id = unrest.4
        title = "[opposition_party] Threaten a Coup?"
        desc = unrest.4.d

        trigger = {
            and = { has_civil_war = no  not = { has_country_flag = during_uprising } }
            not = { has_idea = idea_coup_timeout }

            and = {
                or = { 
                    check_variable = { highest_party_ideology@exclude_ruling_party = token:communism }
                    check_variable = { highest_party_ideology@exclude_ruling_party = token:monarchism }
                }
                or = {
                    check_variable = { highest_party_popularity@exclude_ruling_party > 0.55 }
                    and = { 
                        check_variable = { party_popularity@ruling_party < 0.45 }   
                        has_stability < 0.4 
                    }
                }
            }
        }
        immediate = { hidden_effect = { add_timed_idea = { idea = idea_coup_timeout months = 6 } } }

        option = {
            name = "We must accept their leadership."
            opposition_coup = yes

            # don't act against self
            ai_chance = { 
                modifier = { 
                    not = { or = { has_idea = idea_communists_organize has_idea = idea_monarchists_organize } } 
                    factor = 0 
                } 
            }
        }
        option = {
            name = "Or what?"
            random = { chance = 80 opposition_civil_war = yes }

            # don't act against self
            ai_chance = { 
                modifier = {
                    or = { has_idea = idea_communists_organize has_idea = idea_monarchists_organize } 
                    factor = 0 
                } 
            }
        }
    }
