country_event = {
    id = usa_mexico.1
    title = "American Volunteers in Mexico"     # among agrarianist troops in the civil war
    desc = usa_mexico.1.d

    trigger = {
        tag = USA
        MEX = { has_civil_war = yes }
    }
    fire_only_once = yes
    mean_time_to_happen = { months = 1 }

    immediate = { hidden_effect = { country_event = { id = usa_mexico.2 days = 14 } } }

    option = {
        name = "Condemn their activity."

        add_stability = 0.05
        random_country = {
            limit = { original_tag = MEX has_government = democratic }
            add_opinion_modifier = { target = USA modifier = small_increase }
        }
    }
    option = {
        name = "It is up to them, what they do."
        add_popularity = { ideology = communism popularity = 0.05 }
    }
}
