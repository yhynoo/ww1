add_namespace = manchukuo_transition

country_event = {
    id = manchukuo_transition.1
    title = "The Transition Agreement"
    desc = manchukuo_transition.1.d

    is_triggered_only = yes

    option = {
        name = "Honor the agreement."
        promote_character = MAN_yuan_shikai

        ai_chance = { base = 2 }
        hidden_effect = { 
            set_country_flag = manchukuo_democratic 
            set_politics = { ruling_party = ROOT elections_allowed = yes }
        }
    }
    option = {
        name = "Maintain the Kuomintang government."
        add_popularity = { ideology = democratic popularity = -0.05 }

        ai_chance = { base = 1 }
        hidden_effect = { set_country_flag = manchukuo_warlord }
    }
}

country_event = {
    id = manchukuo_transition.2
    title = "Empire Once Again?"
    desc = manchukuo_transition.2.d

    trigger = {
        has_country_leader = { character = MAN_yuan_shikai ruling_only = yes }
        has_government = democratic
        democratic < 0.5
    }
    fire_only_once = yes

    option = {
        name = "Long live the Hongxian Emperor!"
        set_cosmetic_tag = MAN_empire

        promote_character = MAN_hongxian_emperor
        set_politics = { ruling_party = monarchism elections_allowed = no }

        hidden_effect = {
            retire_character = MAN_yuan_shikai
            promote_character = MAN_tang_shaoyi
        }

        ai_chance = { base = 4 }
    }
    option = {
        name = "We must not abandon the democratic path."
        add_popularity = { ideology = democratic popularity = 0.05 }

        ai_chance = { base = 1 }
    }
}
