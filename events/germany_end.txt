add_namespace = germany_end

country_event = {
    id = germany_end.1
    title = "Opposition Demands Armistice!"
    desc = germany_end.1.d

    trigger = {
        original_tag = GER
        or = {
            and = { not = { controls_state = 17 } not = { controls_state = 51 } }
            surrender_progress > 0.15
        }
    }
    fire_only_once = yes
    mean_time_to_happen = { days = 14 }
    immediate = { 
        hidden_effect = { 
            news_event = { id = germany_end.2 days = 7 random_days = 7 }
            news_event = { id = germany_end.3 days = 21 random_days = 14 }
        } 
    }

    option = {
        name = "This war is lost."
        every_enemy_country = {
            limit = { or = { is_faction_leader = yes is_in_faction = no } }
            country_event = { id = _offer_faction_leader.1 hours = 6 random_days = 7 }
        }
        ai_chance = { 
            base = 2
            modifier = { has_government = democratic factor = 3 }
        }
    }

    option = {
        name = "We shall continue fighting."

        add_popularity = { ideology = ROOT popularity = -0.1 }
        add_political_power = -75

        ai_chance = { 
            base = 1
            modifier = { has_war_support > 0.5 factor = 2 }
        }
    }
}
