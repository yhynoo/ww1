add_namespace = italy_influence_the_uk

## Mediate South Tirol

country_event = {
    id = italy_influence_the_uk.1
    title = "Italy Requests British Mediation on South Tirol"
    desc = italy_influence_the_uk.1.d

    is_triggered_only = yes

    option = {
        name = "Mediate the dispute with Austria."
        AUS = { country_event = { id = italy_influence_the_uk.2 hours = 12 random_hours = 6 } }
    }

    option = {
        trigger = { not = { is_ally_with = ITA } }

        name = "Offer Italy a guarantee instead."
        ITA = { news_event = { id = italy_influence_the_uk.5 hours = 12 random_hours = 6 } }

        give_guarantee = ITA
    }

    option = {
        trigger = { is_ally_with = ITA }

        name = "We won't intervene in this matter."
        ITA = { news_event = { id = italy_influence_the_uk.6 hours = 12 random_hours = 6 } }
    }
}

country_event = {
    id = italy_influence_the_uk.2
    title = "Settlement Over South Tirol?"
    desc = italy_influence_the_uk.2.d

    is_triggered_only = yes

    option = {
        name = "Accept and cede South Tirol to Italy."
        ITA = { news_event = { id = italy_influence_the_uk.3 hours = 12 random_hours = 6 } }

        39 = { transfer_state_to = ITA }        
        850 = { transfer_state_to = ITA }

        ai_chance = {
            base = 1
            modifier = { ITA = { is_ally_with = ENG } factor = 3 }
            modifier = { ITA = { has_country_flag = invited_uk_experts } add = 1 }
        }
    }

    option = {
        name = "Refuse British mediation."
        ITA = { 
            news_event = { id = italy_influence_the_uk.4 hours = 12 random_hours = 6 }
            create_wargoal = {
                target = AUS
                type = take_claimed_state
            }
        }

        ai_chance = {
            base = 2
            modifier = { AUS = { or = { is_ally_with = GER is_ally_with = FRA } } factor = 3 }
        }
    }
}
