    id = sov_agitation.1
    title = "Agitation in the Army"
    desc = sov_agitation.1.d

    trigger = {
        tag = SOV
        and = { surrender_progress > 0.05 not = { has_country_flag = agitation_event_done } }

country_event = {
    id = sov_agitation.3
    title = "Lenin’s Arrival"
    desc = sov_agitation.3.d

    trigger = { 
        tag = SOV
        and = { has_war_with = GER or = { surrender_progress > 0.15 longest_war_length > 24 } }

        has_civil_war = no
        not = { has_government = communism }
    }

    immediate = {
        set_country_flag = agitation_event_done
        hidden_effect = {
            random = { chance = 50 news_event = { id = sov_agitation.2 days = 14 random_days = 7 } }
            random = { chance = 50 news_event = { id = sov_agitation.3 days = 28 random_days = 7 } }
        }
    }

    option = {
        name = "Don’t agitate the workers."

        promote_character = SOV_vladimir_lenin
        add_popularity = { ideology = communism popularity = 0.05 }

        ai_chance = { factor = 1 }
    }
    option = {
        name = "Arrest him."
        add_stability = -0.05
    }
}

    desc = sov_end.1.d

    trigger = {

        and = {
            not = { has_country_flag = end_event_done }
            and = { communism > 0.4 surrender_progress > 0.05 not = { has_government = communism } has_civil_war = no }
        }
    }
    immediate = { set_country_flag = end_event_done }

    option = {
        name = "To arms!"
        start_civil_war = { ideology = communism size = 0.6 }

            soviet_civil_war = yes

        name = "Long live the workers’ state!"

        hidden_effect = { 
            soviet_civil_war = yes
            news_event = { id = sov_end.2 hours = 6 }
        }      
    }
}

# - ukraine
add_namespace = sov_ukraine

country_event = {
    id = sov_ukraine.1
    title = "Arson in Sevastopol"
    desc = sov_ukraine.1.d

    trigger = { 
        tag = SOV
        and = { has_war = no has_stability < 0.25 not = { has_completed_focus = SOV_open_the_economy } not = { has_country_flag = ukraine_event_done } }
        
        and = {
            UKR = { exists = no }
        }
    }
    immediate = { set_country_flag = ukraine_event_done }

    option = {
        name = "Send in the troops."
        add_stability = 0.025

        hidden_effect = { 
            random = { chance = 50 news_event = { id = sov_ukraine.2 days = 7 random_days = 7 } }
        }

        ai_chance = {
            factor = 0
            modifier = { has_country_flag = orthodox_direction add = 1 }
        }
    }
    option = {
        name = "We must not overreact."
        add_popularity = { ideology = democratic popularity = 0.025 }

        ai_chance = {
            factor = 0
            modifier = { has_country_flag = democratic_direction add = 1 }
        }
    }
}
