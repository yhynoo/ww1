add_namespace = austria_end

country_event = {
    id = austria_end.1
    title = "Hungarians Declare Independence!"
    desc = austria_end.1.d

    trigger = {
        tag = AUS
        or = {
            has_idea = idea_shame_of_lemberg
            and = { has_stability < 0.5 surrender_progress > 0.25 }
        }
        not = { has_completed_focus = AUS_embrace_modernity }
    }
    fire_only_once = yes

    immediate = {
        hidden_effect = {
            if = {
                limit = { has_idea = idea_dual_monarchy }
                remove_ideas = idea_dual_monarchy
            }
            if = {
                limit = {
                    any_country = {
                        original_tag = SOV
                        has_government = communism
                    }
                    HUN = {
                        set_popularities = { communism = 45 democratic = 30 monarchism = 25 }
                        set_politics = { ruling_party = communism elections_allowed = no }
                    }
                }
            }
        }
    }

    option = {
        name = "We can do nothing about this now."
        
        release = HUN
        hidden_effect = {
            every_owned_state = {
                limit = { or = { state = 103 state = 104 state = 109 } }
                set_state_owner_to = HUN
            }

            every_controlled_state = {
                limit = { or = { state = 103 state = 104 state = 109 } }
                set_state_controller_to = HUN
            }
        }

        hidden_effect = { HUN = { load_oob = "HUN_1912" } }
        ai_chance = { base = 1 }
    }
    option = {
        name = "Never!"

        hidden_effect = { 
            hungarian_independence = yes
            every_owned_state = {
                limit = { or = { state = 103 state = 104 state = 109 } }
                add_claim_by = HUN
            }
        }
        declare_war_on = { target = HUN type = annex_everything }
        ai_chance = { base = 2 }
    }
}

country_event = {
    id = austria_end.2
    title = "Local Administration in Galicia-Lodomeria"
    desc = austria_end.2.d

    trigger = { 
        tag = AUS
        has_war_with = HUN

        POL = { exists = no }
        any_state = {
            or = { state = 80 state = 89 state = 91 }
            is_owned_and_controlled_by = AUS
        }
    }
    fire_only_once = yes

    immediate = {
        hidden_effect = {
            every_state = {
                limit = {
                    or = { state = 80 state = 89 state = 91 }
                    is_owned_and_controlled_by = AUS
                }
                transfer_state_to = POL
            }
            POL = { 
                set_cosmetic_tag = POL_galicia
                country_event = { id = poland_committee.1 hours = 12 }

                set_capital = { state = 91 }
                load_oob = "GLO_1912"
            }

            if = {
                limit = { has_idea = idea_unrest_in_galicia }
                remove_ideas = idea_unrest_in_galicia
            }
        }
    }

    option = {
        name = "Let them have it."

        news_event = { id = austria_end.3 hours = 6 }
        add_stability = 0.05
    }
    option = {
        name = "These are lands of the crown."
        declare_war_on = { target = POL type = take_claimed_state }
    }
}
