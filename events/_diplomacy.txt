# territory
    country_event = {
        id = diplomacy.6
        title = "[FROM.GetName] Offers Territorial Adjustments"
        desc = diplomacy.6.d

        is_triggered_only = yes

        option = {
            name = "This is a welcome initiative."

            every_state = {
                limit = { 
                    and = { is_controlled_by = FROM not = { is_claimed_by = FROM } }
                    is_core_of = ROOT
                }
                transfer_state_to = ROOT
            }
            every_state = {
                limit = {
                    and = { is_controlled_by = ROOT not = { is_claimed_by = ROOT } }
                    is_core_of = FROM
                }
                transfer_state_to = FROM
            }

            # effect news
            FROM = { news_event = { id = shared.22 hours = 6 } }
        }
        option = {
            name = "We can’t hold these areas now."
        }
    }

# armistice
    country_event = {
        id = diplomacy.7
        title = "Time for an Armistice?"
        desc = diplomacy.7.d

        trigger = {
            has_war = yes
            has_civil_war = no

            is_subject = no
            has_war_support < 0.25
            longest_war_length > 6

            any_enemy_country = {
                or = { is_faction_leader = yes is_in_faction = no }     # any important enemy is undamaged
                surrender_progress < 0.05
            }

            not = { has_idea = idea_armistice_timeout }
        }
        immediate = { hidden_effect = { add_timed_idea = { idea = idea_armistice_timeout days = 180 } } }

        option = {
            name = "It is time."
            every_enemy_country = {
                limit = { or = { is_faction_leader = yes is_in_faction = no is_subject = no } }
                country_event = { id = diplomacy.8 hours = 12 }
            }
        }
        option = {
            name = "We shall fight until victory."
            add_political_power = -100
        }
    }

    country_event = {
