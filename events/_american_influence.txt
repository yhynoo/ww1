add_namespace = _american_influence

country_event = {
    id = _american_influence.1
    title = "American Investments"
    desc = _american_influence.1.d

    is_triggered_only = yes

    option = {
        name = "Why not."

        every_controlled_state = {
            limit = { check_variable = { investment = 1 } }
            custom_effect_tooltip = tt_allow_investment
        }

        hidden_effect = { 
            USA = { activate_decision = investment_in_progress } 
        }
    }
    option = {
        name = "No."
        FROM = { news_event = { id = _american_influence.2 hours = 6 } }

        hidden_effect = {
            every_controlled_state = {
                limit = { check_variable = { investment = 1 } }

                # clear the variable.
                set_variable = { investment = 0 }
            }
        }
    }
}

# expelling

news_event = {
    id = _american_influence.3
    title = "[ROOT.GetName] Expels American Investors!"
    desc = _american_influence.3.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "Interesting."

        if = { 
            limit = { tag = USA } 
            add_opinion_modifier = { target = ROOT modifier = opinion_expelled_american_investors }
        }
    }
}

# fade

news_event = {
    id = _american_influence.4
    title = "Our Influence in Central America Fades"
    desc = _american_influence.4.d

    trigger = {
        tag = USA
        not = {
            any_state = {
                or = { is_controlled_by = MEX is_controlled_by = COL }
                has_resources_rights = { receiver = USA }
            }
        }
    }
    fire_only_once = yes

    option = {
        name = "Disappointing."

        add_political_power = -100
        add_popularity = { ideology = ROOT popularity = -0.05 }

        #
        remove_ideas = idea_influence_in_central_america
    }
}
