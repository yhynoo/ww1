## -- oil 
add_namespace = aus_oil

country_event = {
    id = aus_oil.1
    title = "[FROM.GetName] Asks for Access to Oil"
    desc = aus_oil.1.d

    is_triggered_only = yes

    option = {
        name = "This is a good idea."
        custom_effect_tooltip = tt_romanian_oil
        
        hidden_effect = { 
            news_event = { id = aus_oil.2 hours = 6 }

            give_resource_rights = { receiver = FROM state = 46 resources = { oil } }
            FROM = { give_guarantee = ROOT } 
        }

        ai_chance = { factor = 1 }
    }
    option = {
        name = "We are not interested."
        FROM = { add_opinion_modifier = { target = ROOT modifier = opinion_denied_access_to_oil } }
        
        ai_chance = { factor = 1 }
    }
}

news_event = {
    id = aus_oil.2
    title = "Oil Concessions for Guarantees"
    desc = aus_oil.2.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "Alright."
        if = { 
            limit = { tag = GER }
            add_opinion_modifier = { target = FROM modifier = opinion_romanian_oil }
        }
    }
}

    id = aus_galicia.1
    title = "Unrest in Galicia"         # suspected Okhrana activity
    desc = aus_galicia.1.d

    is_triggered_only = yes

    option = {
        name = "Worrying."
        add_timed_idea = { idea = idea_unrest_in_galicia months = 6 }

        hidden_effect = {
            random = { chance = 50 news_event = { id = aus_galicia.2 days = 14 random_days = 7 } }
            country_event = { id = aus_galicia.3 days = 42 random_days = 14 }
        }
    }
}

news_event = {
    id = aus_galicia.2
    title = "Desertions at the Tarnopol Garrison"
    desc = aus_galicia.2.d

    is_triggered_only = yes
    trigger = { controls_state = 91 }

    option = {
        name = "This is unfortunate."
        add_war_support = -0.025
    }
}

country_event = {
    id = aus_galicia.3
    title = "Archduke Assassinated in Lemberg!"
    desc = aus_galicia.3.d

    is_triggered_only = yes
    trigger = {
        controls_state = 91
        and = { has_government = monarchism not = { has_war_with = SOV } }   
    }
    immediate = {
        hidden_effect = {
            SOV = { add_named_threat = { name = "Archduke Assassinated in Lemberg" threat = 12 } }
            retire_character = AUS_franz_ferdinand
            news_event = { id = aus_archduke.1 hours = 6 }
        }
    }

    option = {
        name = "We must avenge his death!"
        add_war_support = 0.25
        declare_war_on = { target = SOV type = topple_government }

        ai_chance = { base = 4 }
    }
    option = {
        name = "What a tragedy."

        add_popularity = { ideology = ROOT popularity = -0.15 }
        ai_chance = { base = 1 }
    }
}
