# oil 
add_namespace = aus_oil

country_event = {
    id = aus_oil.1
    title = "[FROM.GetName] Asks for Access to Oil"
    desc = aus_oil.1.d

    is_triggered_only = yes

    option = {
        name = "This is a good idea."
        custom_effect_tooltip = tt_romanian_oil
        
        hidden_effect = { 
            news_event = { id = aus_oil.2 hours = 6 }

            give_resource_rights = { receiver = FROM state = 46 resources = { oil } }
            FROM = { give_guarantee = ROOT } 
        }

        ai_chance = { 
            factor = 1 
            modifier = { or = { is_guaranteed_by = GRE is_guaranteed_by = SER } factor = 0 }
        }
    }
    option = {
        name = "We are not interested."
        FROM = { add_opinion_modifier = { target = ROOT modifier = opinion_denied_access_to_oil } }
        
        ai_chance = { factor = 1 }
    }
}

# unrest among the slavs
add_namespace = aus_unrest

news_event = {
    id = aus_unrest.1
    title = "Unrest Among the Slavic Subjects"
    desc = aus_unrest.1.d

    trigger = {
        tag = AUS
        not = { has_country_flag = unrest_event_done }
        has_government = monarchism
        
        date > 1914.1.1
        not = { has_war_with = SER }
        and = { SOV = { has_completed_focus = SOV_deterrence } SER = { has_completed_focus = SER_deterrence } }
    }
    mean_time_to_happen = { days = 14 }
    
    immediate = { 
        set_country_flag = unrest_event_done
        hidden_effect = { 
            add_named_threat = { threat = 2 name = "Unrest Among Slavs in Austria-Hungary" } 
            random = { chance = 50 news_event = { id = aus_unrest.2 days = 28 random_days = 7 } }
            country_event = { id = aus_unrest.3 days = 70 random_days = 70 }
        } 
    }

    option = {
        name = "Awful."
        add_timed_idea = { idea = idea_slavic_unrest years = 1 }
    }
}
