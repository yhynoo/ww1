# - sarajevo
add_namespace = aus_sarajevo

country_event = {
    id = aus_sarajevo.1
    title = "Annexation of Sarajevo"        # didn't happen in 1908 in this timeline
    desc = aus_sarajevo.1.d

    fire_only_once = yes
    trigger = {
        tag = AUS
        date > 1912.5.1
        controls_state = 104
    }

    option = {
        name = "This is the logical thing to do."
        104 = { add_core_of = AUS }

        hidden_effect = { 
            add_named_threat = { name = "[AUS.GetName] Annexes Sarajevo" threat = 4 }
            TUR = { add_opinion_modifier = { target = ROOT modifier = opinion_integrated_sarajevo } }
        }
        ai_chance = { factor = 4 }
    }
    option = {
        name = "We must not agitate the Turks."
        ai_chance = { factor = 1 }
    }
}

## -- unrest in galicia
add_namespace = aus_galicia

news_event = {
    id = aus_galicia.1
    title = "Unrest in Galicia"         # suspected Okhrana activity
    desc = aus_galicia.1.d

    is_triggered_only = yes

    option = {
        name = "Worrying."
        add_timed_idea = { idea = idea_unrest_in_galicia months = 6 }

        hidden_effect = {
            random = { chance = 50 news_event = { id = aus_galicia.2 days = 14 random_days = 7 } }
            country_event = { id = aus_galicia.3 days = 42 random_days = 14 }
        }
    }
}

news_event = {
    id = aus_galicia.2
    title = "Desertions at the Tarnopol Garrison"
    desc = aus_galicia.2.d

    is_triggered_only = yes
    trigger = { controls_state = 91 }

    option = {
        name = "This is unfortunate."
        add_war_support = -0.025
    }
}

country_event = {
    id = aus_galicia.3
    title = "Archduke Assassinated in Lemberg!"
    desc = aus_galicia.3.d

    is_triggered_only = yes
    trigger = {
        controls_state = 91
        and = { has_government = monarchism not = { has_war_with = SOV } }   
    }
    immediate = {
        hidden_effect = {
            SOV = { add_named_threat = { name = "Archduke Assassinated in Lemberg" threat = 12 } }
            retire_character = AUS_franz_ferdinand
            news_event = { id = aus_archduke.1 hours = 6 }
        }
    }

    option = {
        name = "We must avenge his death!"
        add_war_support = 0.25
        declare_war_on = { target = SOV type = topple_government }

        ai_chance = { base = 4 }
    }
    option = {
        name = "What a tragedy."

        add_popularity = { ideology = ROOT popularity = -0.15 }
        ai_chance = { base = 1 }
    }
}
