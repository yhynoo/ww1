add_namespace = turkey_military_elite

country_event = {
    id = turkey_military_elite.1
    title = "Old Elites Offer Financial Contributions"
    desc = turkey_military_elite.1.d

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            country_event = { id = turkey_military_elite.2 days = 14 random_days = 14 }
        }
    }

    option = {
        name = "Their help is most welcome."

        add_political_power = -150
        if = {
            limit = { has_idea = sww1_idea_amassed_foreign_debt }
            swap_ideas = {
                remove_idea = sww1_idea_amassed_foreign_debt
                add_idea = sww1_idea_significant_foreign_debt
            }
        }
        else_if = {
            limit = { has_idea = sww1_idea_significant_foreign_debt }
            remove_ideas = sww1_idea_significant_foreign_debt
        }
        else = {
            random_owned_controlled_state = {
                add_building_construction = {
                    type = industrial_complex
                    level = 1
                    instant_build = yes
                }
            }
        }

        ai_chance = { base = 3 }
    }
    option = {
        name = "This is not needed."
        ai_chance = { base = 1 }
    }
}
