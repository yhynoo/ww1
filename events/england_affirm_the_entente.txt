### 2. Country Event: Britain Invites [FROM.GetName] to the Entente
country_event = {
    id = britain_affirm_the_entente.2
    title = "Britain Invites [From.GetName] to the Entente"
    desc = britain_affirm_the_entente.2.d

    is_triggered_only = yes

    option = {
        name = "Join the Entente."
        ENG = { news_event = { id = britain_affirm_the_entente.3 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 2
            modifier = { or = { threat > 0.1 GER = { is_in_faction = yes } } factor = 3 }
        }
    }

    option = {
        name = "Decline the invitation."
        ENG = { news_event = { id = britain_affirm_the_entente.4 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 1
            modifier = { has_completed_focus = FRA_ancien_regime factor = 12 }
        }
    }
}
