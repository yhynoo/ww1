add_namespace = germany_look_to_the_east

### 1. Germany Proposes an Alliance to the Qing Empire
country_event = {
    id = germany_look_to_the_east.1
    title = "Germany Proposes an Alliance"
    desc = germany_look_to_the_east.1.d

    is_triggered_only = yes

    option = {
        name = "Accept the alliance offer."

        GER = { 
            news_event = { id = germany_look_to_the_east.2 hours = 12 random_hours = 6 }
            if = {
                limit = { is_in_faction = no }
                create_faction = "Central Powers"
            }
            faction_leader = { add_to_faction = MAN }
        }

        ai_chance = { 
            base = 2
            modifier = {
                JAP = { is_in_faction = yes }
                factor = 6
            }
        }
    }

    option = {
        name = "Reject the alliance offer."
        GER = { news_event = { id = germany_look_to_the_east.3 hours = 12 random_hours = 6 } }

        ai_chance = { base = 1 }
    }
}
