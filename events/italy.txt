### 1. Italy Demands Libya
country_event = {
    id = italy_events.1
    title = "Italian Demands for Libya"
    desc = italy_events.1.d

    is_triggered_only = yes

    option = {
        name = "Yield to Italian demands."
        news_event = { id = italy_events.2 hours = 12 random_hours = 6 }

        every_state = {
            limit = {
                or = {
                    state = 448
                    state = 449
                    state = 450
                    state = 451

                    state = 661
                    state = 662
                    state = 663

                    state = 273
                }
                is_owned_by = TUR
            }
            transfer_state_to = ITA
        }

        ai_chance = {
            base = 2
        }
    }

    option = {
        name = "Resist Italian aggression."
        ITA = { 
            country_event = { id = italy_events.3 hours = 12 random_hours = 6 }
            create_wargoal = {
                target = TUR
                type = take_claimed_state 
            }
        }

        ai_chance = {
            base = 1
            modifier = {
                any_country = {
                    is_major = yes
                    has_guaranteed = TUR
                }
                factor = 5
            }
            modifier = {
                is_in_faction = 10
            }
        }
    }
}
