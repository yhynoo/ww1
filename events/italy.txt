# - tripoli
add_namespace = ita_tripoli

country_event = {
    id = ita_tripoli.1
    title = "After the Defeat of Tripoli"
    desc = ita_tripoli.1.d

    is_triggered_only = yes

    option = {
        name = "What traitor celebrates a defeat?"
        add_war_support = 0.02
        
        add_popularity = { ideology = monarchism popularity = 0.025 }
        set_country_flag = monarchist_orientation

        ai_chance = { factor = 3 }
    }
    option = {
        name = "We told you so."

        add_popularity = { ideology = communism popularity = 0.05 }
        set_country_flag = socialist_orientation

        ai_chance = { factor = 1 }
    }
}

    trigger = { has_stability < 0.5 }

    option = {
        name = "Send in the troops."
        add_stability = 0.025

        ai_chance = {
            factor = 0
            modifier = { has_country_flag = monarchist_orientation add = 1 }

        name = "Ignore it."
        add_popularity = { ideology = communism popularity = 0.05 }

        ai_chance = {
            factor = 0
            modifier = { has_country_flag = socialist_orientation add = 1 }
        }
    }
}
