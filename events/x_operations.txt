add_namespace = x_operations

country_event = {
    id = x_operations.1
    title = "'Through Metz to Paris'"
    desc = x_operations.1.d
    
    trigger = {
        tag = GER
        has_war_with = FRA
    }
    fire_only_once = yes

    option = {
        name = "Execute the plan."
        add_timed_idea = { idea = idea_metz_to_paris months = 3 }

        hidden_effect = { country_event = { id = x_operations.2 months = 2 days = 21 } }

        ai_chance = { base = 1 }
    }
    option = {
        name = "The situation does not warrant it now."
        army_experience = 50

        ai_chance = { base = 0 }
    }
}

country_event = {

}
