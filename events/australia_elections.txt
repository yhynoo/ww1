    title = "The Election of 1914"
    desc = australia_elections.1.d

    is_triggered_only = yes

        tag = AST
        has_government = democratic

        date > 1914.1.1         date < 1915.1.1

    option = {
        name = "Andrew Fisher."

            limit = { is_subject = yes }

        }
        else = { 
            add_war_support = 0.05
        }

        promote_character = AST_joseph_cook
        ai_chance = { base = 1 }
    }
}
