add_namespace = manchukuo_elections

country_event = {
    id = manchukuo_elections.1
    title = "The Election of 1913"
    desc = manchukuo_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = MAN
        has_government = democratic
        not = { has_idea = idea_uneasy_agreement }

        date > 1913.1.1         date < 1914.1.1
    }

    option = {
        name = "Sun Yat-Sen."
        trigger = { 
            has_country_leader = { character = MAN_sun_yat_sen }
            not = { has_idea = idea_sun_in_exile } 
        }

        if = {
            limit = { has_country_leader = { character = MAN_sun_yat_sen ruling_only = yes } }
            add_stability = 0.05
        }
        else = {
            add_popularity = { ideology = democratic popularity = 0.025 }
            promote_character = MAN_sun_yat_sen
        }

        ai_chance = { 
            base = 0
            modifier = { has_idea = idea_aborted_transition add = 1 }
        }
    }
    option = {
        name = "Yuan Shikai."
        if = {
            limit = { has_country_leader = { character = MAN_yuan_shikai ruling_only = yes } }
            add_stability = 0.05
        }
        else = {
            add_war_support = 0.05
            promote_character = MAN_yuan_shikai
        }

        ai_chance = { 
            base = 1
            modifier = { has_idea = idea_aborted_transition factor = 0 }
        }
    }
}
