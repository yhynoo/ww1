add_namespace = germany_baltic

country_event = {
    id = germany_baltic.1
    title = "[GER.GetName] Demands Baltic Independence"
    desc = germany_baltic.1.d

    is_triggered_only = yes

    option = {
        name = "We must agree to their demands."

        release = UBD
        GER = { puppet = UBD }

        hidden_effect = { news_event = { id = germany_baltic.2 days = 7 random_days = 7 } }
    }
    option = {
        name = "How about an alliance instead?"

        trigger = {
            is_in_faction = no
            GER = { or = { is_in_faction = no is_faction_leader = yes } }
        }

        GER = { country_event = { id = germany_baltic.3 days = 7 random_days = 7 } }
    }
    option = {
        name = "Who do they think they are?"
        GER = { 
            news_event = { id = germany_baltic.6 hours = 6 random_hours = 6 }
            create_wargoal = {
                target = ROOT
                type = liberate_wargoal
            }
        }
    }
}
