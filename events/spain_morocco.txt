    title = "Support the French Intervention?"
    desc = spain_morocco.1.d

    is_triggered_only = yes

    option = {
        name = "We have enough troubles here."
        add_war_support = -0.05

        ai_chance = { base = 2 }
    }

    option = {
        name = "This is the right thing to do."

        declare_war_on = { target = MOR type = annex_everything }

        hidden_effect = {
            set_country_flag = pro_french

    }

    option = {
        name = "Support the Moroccans instead."
        send_equipment = { equipment = infantry_equipment amount = 600 target = MOR }

        hidden_effect = {
            set_country_flag = anti_french
        }

        ai_chance = { base = 1 }
    }
}
