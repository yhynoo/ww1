add_namespace = _the_great_war

news_event = {
    id = _the_great_war.0
    hidden = yes

    trigger = {
        GER = {
            and = {
                or = { 
                    has_war_with = ENG 
                    has_war_with = FRA
                }
                or = {
                    or = { is_ally_with = AUS has_war_with = AUS }
                    or = { is_ally_with = TUR has_war_with = TUR }
                }
            }
        }
    }
    fire_only_once = yes

    immediate = {
        news_event = _the_great_war.1
        set_global_flag = the_great_war_has_started

        every_country = {
            limit = { 
                or = {
                    tag = GER 
                    is_ally_with = GER 
                    has_war_with = GER
                }
            }
            
            # news
            set_country_flag = side_of_the_great_war
            news_event = { id = _the_great_war.2 days = 7 }
        }
    }
}

# START

news_event = {
    id = _the_great_war.1
    title = "Great War in Europe!"
    desc = _the_great_war.1.d

    is_triggered_only = yes 
    major = yes

    option = {
        name = "A war between Europeans is a civil war."
    }
}

news_event = {
    id = _the_great_war.2
    title = "The Spirit of [GetYear]"
    desc = _the_great_war.2.d

    is_triggered_only = yes
    trigger = {
        not = { has_idea = idea_shame_of_lemberg }
        not = { has_idea = idea_widespread_hunger }
        not = { has_idea = idea_army_mutinies }
        not = { has_idea = idea_spanish_flu }
    }

    option = {
        name = "Avenge Agadir!"
        trigger = { 
            original_tag = ENG 
            has_war_with = GER
        }

        add_war_support = 0.1
        add_timed_idea = { idea = idea_war_enthusiasm months = 6 }
    }

    option = {
        name = "France d'abord!"
        trigger = { 
            original_tag = FRA 
            has_war_with = GER
        }

        add_war_support = 0.1
        add_timed_idea = { idea = idea_war_enthusiasm months = 6 }
    }

    option = {
        name = "Für Gott und Vaterland!"
        trigger = { original_tag = GER }

        add_war_support = 0.1
        add_timed_idea = { idea = idea_war_enthusiasm months = 6 }
    }    

    option = {
        name = "Advance!"
        trigger = {
            not = { and = { original_tag = ENG has_war_with = GER } }
            not = { and = { original_tag = FRA has_war_with = GER } }
            not = { original_tag = GER }
        }

        add_war_support = 0.05
        add_timed_idea = { idea = idea_war_enthusiasm months = 6 }
    }
}
