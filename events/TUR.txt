# END THE WAR
    country_event = {
        id = turkey.5
        title = "End the War?"
        desc = turkey.5.d

        trigger = {
            and = { tag = TUR has_country_flag = southeastern_europe_war }
            war_length_with = { tag = ROM   months > 3 }
            has_war_support < 0.5
        }
        fire_only_once = yes

        option = {
            name = "We cannot continue."

            # consequence event
            every_country = {
                limit = { or = { tag = GRE tag = SER tag = ROM } has_war_with = TUR }
                country_event = { id = turkey.6 hours = 6 }

        }
        option = {
            name = "[TUR.GetNameDefCap] must endure."

            # block non-historical
            ai_chance = { modifier = { is_historical_focus_on = yes factor = 0 } }
        }
    }
