add_namespace = ukraine_uprising

news_event = {
    id = ukraine_uprising.1
     title = "Unrest Among Ukrainian Peasants"       
    desc = ukraine_uprising.1.d 

    trigger = {
        any_country = {
            and = { 
                original_tag = SOV controls_state = 202 
                has_stability < 0.25
                or = { has_war = no has_civil_war = yes }
            }
        }
        UKR = { exists = no }
    }
    fire_only_once = yes

    option = {
        name = "Worrying."
        add_timed_idea = { idea = idea_unrest_in_ukraine months = 6 }
        hidden_effect = { 
            random = {
                chance = 25
                news_event = { id = ukraine_uprising.2 days = 14 random_days = 60 } 
            }
        }
    }
}

news_event = {
    id = ukraine_uprising.2
    title = "Uprising in the Steppe!"
    desc = ukraine_uprising.2.d 

    is_triggered_only = yes
    trigger = { 
        has_stability < 0.5 
        or = { has_war = no has_civil_war = yes }
        
        UKR = { exists = no }
        not = { has_completed_focus = SOV_open_the_economy }
    }

    option = {
        name = "To arms!"
        remove_ideas = idea_unrest_in_ukraine

        every_country = {
            limit = { original_tag = SOV }

            hidden_effect = { ukrainian_independence = yes }
            declare_war_on = { target = UKR type = annex_everything }
        }
    }
}
