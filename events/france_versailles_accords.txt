add_namespace = france_versailles_accords

country_event = {
    id = france_versailles_accords.1
    title = "Affirm Our Alliance with [SOV.GetNameDef]?"
    desc = france_versailles_accords.1.d

    trigger = {
        tag = FRA
        has_war_with = GER

        is_in_faction = no
        SOV = { is_in_faction = no }

        not = { has_country_flag = no_alliance_with_russia }
        SOV = { or = { has_government = monarchism has_government = democratic } }
    }

    option = {
        name = "This will be useful."
        SOV = { country_event = { id = france_versailles_accords.2 hours = 12 random_hours = 6 } }

        ai_chance = { base = 6 }
    }

    option = {
        name = "We don't need this."
        set_country_flag = no_alliance_with_russia

        ai_chance = { base = 1 }
    }
}

    }
}
