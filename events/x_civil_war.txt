# truce

country_event = {
    id = x_civil_war.1
    title = "[FROM.GetName] Offers Truce"
    desc = x_civil_war.1.d

    is_triggered_only = yes

    option = {
        name = "This is a chance to focus on our internal affairs."

        FROM = { 
            news_event = { id = x_civil_war.2 hours = 6 random_hours = 6 }
            every_state = {
                limit = {
                    is_owned_by = ROOT is_controlled_by = FROM        # all ROOT's states that FROM conquered
                }
                transfer_state_to = FROM
            }
        }

        white_peace = FROM
        ai_chance = { base = 2 }
    }
    option = {
        name = "We will never agree to this."
        FROM = { 
            news_event = { id = x_civil_war.3 hours = 6 random_hours = 6 }
            add_war_support = 0.05
        }

        ai_chance = { base = 1 }
    }
}
