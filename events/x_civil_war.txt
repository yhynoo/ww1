    id = x_civil_war.1
    title = "[FROM.GetName] Offers Truce"
    desc = x_civil_war.1.d

    is_triggered_only = yes

    option = {
        name = "Let's agree for now."
        white_peace = FROM

        FROM = {
            every_core_state = {
                limit = { is_controlled_by = FROM owns_state = ROOT }
                transfer_state_to = FROM
            }
        }
        
        ai_chance = { base = 3 }
    }
    option = {
        name = "We are not interested."
        ai_chance = { base = 1 }
    }
}

country_event = {
    id = x_civil_war.4
    title = "[FROM.GetName] Demands Territories"
    desc = x_civil_war.4.d

    is_triggered_only = yes

    option = {
        name = "They are rightfully theirs."
        FROM = { news_event = { id = x_civil_war.5 hours = 6 random_hours = 6 } }

        every_controlled_state = {
            limit = { or = { is_core_of = FROM is_claimed_by = FROM } }
            transfer_state_to = FROM
        }
        
        ai_chance = { 
            base = 2
            modifier = { has_government = communism add = 3 }
        }
    }
    option = {
        name = "So what?"
        ai_chance = { base = 1 }

        FROM = { 
            news_event = { id = x_civil_war.6 hours = 6 random_hours = 6 }
            create_wargoal = { target = ROOT type = take_state_focus }
        }
    }
}

country_event = {
    id = x_civil_war.7
    title = "[FROM.GetName] Demands Withdrawal"
    desc = x_civil_war.7.d

    is_triggered_only = yes

    option = {
        name = "We no longer need to occupy those lands."
        FROM = { news_event = { id = x_civil_war.8 hours = 6 random_hours = 6 } }

        every_controlled_state = {
            limit = { is_core_of = FROM not = { is_core_of = ROOT } }
            transfer_state_to = FROM
        }
        
        ai_chance = { 
            base = 1
            modifier = { has_war = yes add = 2 }

            # france won't every give anything to germans.
            modifier = { and = { ROOT = { tag = FRA } FROM = { original_tag = GER } } factor = 0 }
        }
    }
    option = {
        name = "We are not interested."
        ai_chance = { base = 1 }

        FROM = { 
            news_event = { id = x_civil_war.9 hours = 6 random_hours = 6 }
            create_wargoal = { target = ROOT type = take_state_focus }
        }
    }
}
