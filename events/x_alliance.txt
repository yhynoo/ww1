add_namespace = x_alliance

# Weak to strong

country_event = {
    id = x_alliance.1
    title = "[FROM.GetName] Asks for Alliance"     
    desc = x_alliance.1.d

    is_triggered_only = yes

    option = {
        name = "We accept their offer."

        if = {
            limit = { is_in_faction = no }
            create_appropriate_faction = yes
        }
        add_to_faction = FROM
            
        FROM = { news_event = { id = x_alliance.3 hours = 6 random_hours = 6 } }

        ai_chance = {
            base = 1

            # if we're at war already, or we lead the faction - then it's more likely we will agree.
            modifier = { or = { is_faction_leader = yes has_war = yes } add = 2 }

            # if we're guaranteeing them: sure
            modifier = { has_guaranteed = FROM factor = 3 }

            # if ENG goes communist: no
            modifier = {
                and = { has_completed_focus = ENG_abolish_lords not = { FROM = { has_government = communism } } }
                factor = 0
            }
        }
    }
    option = {
        name = "We are not interested."
        FROM = { news_event = { id = x_alliance.4 hours = 6 random_hours = 6 } }

        ai_chance = { base = 1 }
    }
}

# Strong to weak

country_event = {
    id = x_alliance.2
    title = "[FROM.GetName] Offers Alliance"     
    desc = x_alliance.2.d

    is_triggered_only = yes

    option = {
        name = "This is a good idea."
        
        FROM = {
            if = {
                limit = { is_in_faction = no }
                create_appropriate_faction = yes
            }
            
            news_event = { id = x_alliance.3 hours = 6 random_hours = 6 }
            faction_leader = { add_to_faction = ROOT }

            ai_chance = {
                base = 2

                # if we're guaranteeing them
                modifier = { FROM = { has_guaranteed = ROOT } factor = 2 }

                modifier = { 
                    and = { FROM = { tag = ENG } tag = FRA has_completed_focus = FRA_adjust_government }
                    factor = 0
                }

                # if we're Turkey and Germany asks
                modifier = {
                    and = { from = { tag = GER } has_completed_focus = TUR_adjust_the_government }
                    factor = 0
                }
            }

            hidden_effect = {
                add_ai_strategy = { type = alliance tag = FROM value = 60 }
            }
        }
    }
    option = {
        name = "We are not interested."
        FROM = { news_event = { id = x_alliance.4 hours = 6 random_hours = 6 } }

        ai_chance = { 
            base = 1

            # if we're France and Austria invites - we won't say no.
            modifier = {
                and = { 
                    tag = FRA 
                    or = {
                        FROM = { tag = AUS }
                        and = { FROM = { tag = ENG } has_completed_focus = FRA_traditionalism }
                    }
                }
                factor = 0
            }

            # if we're at war, we won't say no.
            modifier = { has_war = yes factor = 0 }
        }
    }
}
