# expel
add_namespace = sau_expel

country_event = {
    id = sau_expel.1
    title = "The Offence of the Turkish Governor"
    desc = sau_expel.1.d

    is_triggered_only = yes
    trigger = { is_subject_of = TUR }

    option = {
        name = "Expel him."
        add_autonomy_score = { value = 120 }

        hidden_effect = { TUR = { news_event = { id = sau_expel.2 hours = 6 } } }
        ai_chance = { factor = 0 }
    }
    option = {
        name = "Let him stay."
        add_political_power = -50
        ai_chance = { factor = 1 }
    }
}

news_event = {
    id = sau_expel.2
    title = "[SAU.GetName] Expels Our Governor"
    desc = sau_expel.2.d

    is_triggered_only = yes

    option = {
        name = "Worrying."
        add_opinion_modifier = { target = FROM modifier = opinion_expelled_governor }
    }
}

# tribalism
add_namespace = sau_tribalism

country_event = {
    id = sau_tribalism.1
    title = "End Arab Tribalism?"
    desc = sau_tribalism.1.d

    is_triggered_only = yes

    option = {
        name = "Announce concessions."
        add_political_power = -50
        add_popularity = { ideology = democratic popularity = 0.05 }
    }
    option = {
        name = "Arrest tribal leaders."
        remove_ideas = idea_tribal_state

        hidden_effect = { 
            random = { chance = 50 news_event = { id = sau_tribalism.2 days = 7 random_days = 7 } }
        }
    }
}

news_event = {
    id = sau_tribalism.2
    title = "Other Tribes Resist Us"
    desc = sau_tribalism.2.d

    is_triggered_only = yes

    option = {
        name = "Not great."

        add_political_power = -100
        random_owned_controlled_state = { limit = { is_core_of = ROOT is_capital = no } remove_core_of = ROOT }
    }
}
