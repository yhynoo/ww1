add_namespace = italy_alps

    id = italy_alps.1
    title = "Italy Annexes Mediterranean Territories"
    desc = italy_alps.1.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "Concerning."
    }
}

country_event = {
    id = italy_alps.2
    title = "Opportunity in the Adriatic"
    desc = italy_alps.2.d

    trigger = {
        tag = ITA
        has_war_support > 0.5

        AUS = { 
            has_war = yes 
            not = { 
                has_war_with = ITA 
                is_ally_with = ITA
            }
        }

        any_state = {
            is_claimed_by = ITA
            is_owned_and_controlled_by = AUS
        }
    }

    fire_only_once = yes
    mean_time_to_happen = { months = 6 }

    option = {
        name = "Demand Tirol and Trieste."
        any_state = {
            or = { state = 33 state = 39 }
            is_owned_and_controlled_by = AUS
        }

        AUS = { country_event = { id = italy_alps.3 hours = 6 random_hours = 6 } }
    }

    option = {
        name = "Extend our claims to Dalmatia."
        trigger = {
            103 = { is_owned_and_controlled_by = AUS }
        }

        103 = { add_claim_by = ITA }
        AUS = { country_event = { id = italy_alps.3 hours = 6 random_hours = 6 } }
    }

    option = {
        name = "It is dishonorable to treat a former ally like this."
        add_war_support = -0.05

        ai_chance = {
            base = 0
            modifier = { has_completed_focus = ITA_adjust_the_government add = 4 }
        }
    }
}

country_event = {
    id = italy_alps.3
    title = "Rome Offers Neutrality for Territories"
    desc = italy_alps.3.d

    is_triggered_only = yes

    option = {
        name = "We must agree."
        ITA = { 
            diplomatic_relation = { country = AUS relation = non_aggression_pact active = yes } 
            hidden_effect = { set_country_flag = neutral_italy }
        }

        every_state = {
            limit = {
                is_owned_and_controlled_by = AUS
                is_claimed_by = ITA
            }
            transfer_state_to = ITA
        }
    }

    option = {
        name = "These disgusting traitors!"
        
        add_war_support = 0.05
        ITA = { create_wargoal = { type = take_claimed_state target = AUS } }
    }
}
