## -- oil 
add_namespace = GER_oil

country_event = {
    id = GER_oil.1
    title = "[FROM.GetName] Asks for Access to Oil"
    desc = GER_oil.1.d

    is_triggered_only = yes

    option = {
        name = "This is a good idea."
        custom_effect_tooltip = tt_iraqi_oil
        
        manage_turkish_debt = yes
        hidden_effect = { 
            news_event = { id = GER_oil.2 hours = 6 }

            give_resource_rights = { receiver = FROM state = 291 resources = { oil } }
            FROM = { give_guarantee = ROOT } 
        }

        ai_chance = {
            factor = 1
            modifier = { or = { has_idea = idea_amassed_foreign_debt has_idea = idea_significant_foreign_debt } factor = 4 }
        }
    }
    option = {
        name = "We are not interested."
        FROM = { add_opinion_modifier = { target = ROOT modifier = opinion_denied_access_to_oil } }
        
        ai_chance = { factor = 1 }
    }
}

    id = GER_oil.2

## -- atlantic
add_namespace = GER_atlantic

country_event = {
    id = GER_atlantic.1
    title = "Atlantic Submarine Warfare"
    desc = GER_atlantic.1.d

    trigger = {
        and = { tag = GER   war_length_with = { tag = ENG months > 6 } }
        not = { has_country_flag = ger_atlantic_done }
    }
    immediate = { set_country_flag = ger_atlantic_done }

    option = {
        name = "Gott Strafe England!"
        add_war_support = 0.05

        swap_ideas = {
            remove_idea = idea_tirpitz_plan
            add_idea = idea_atlantic_submarine_warfare
        }

        hidden_effect = {
            random = { chance = 50 USA = { news_event = { id = GER_atlantic.2 days = 28 random_days = 140 } } }

            # -- argentina and spain
            every_country = {
                limit = { or = { tag = ARG tag = SPA } }
                random = { chance = 50 news_event = { id = GER_atlantic.3 days = 28 random_days = 140 } }
            }
        }

        ai_chance = { factor = 4 }
    }
    option = {
        name = "We will win in a conventional way."
        ai_chance = { factor = 1 }
    }
}
news_event = {
    id = GER_atlantic.2
    title = "Lusitania Lost in a Submarine Attack!"

    is_triggered_only = yes
    major = yes

    option = { 
        name = "What a tragedy."

            add_timed_idea = { idea = idea_lusitania months = 6 }
            add_opinion_modifier = { target = GER modifier = opinion_sinking_lusitania }
        }
    }
}

## -- 1916
add_namespace = GER_harvest

country_event = {
    id = GER_harvest.1
    title = "The Harvest of 1916"
    desc = GER_harvest.1.d

    trigger = {
        and = { tag = GER       date > 1916.7.1     date < 1916.10.1 }
        not = { has_idea = civilian_economy }
    }
    fire_only_once = yes

    option = {
        name = "Demobilize the economy."
        custom_effect_tooltip = tt_demobilize_economy
        hidden_effect = {
            if = {
                limit = { has_idea = idea_rationing }
                remove_ideas = idea_rationing
            }
            add_timed_idea = { idea = idea_harvest_1916 months = 6 }
            add_ideas = civilian_economy
        }
        ai_chance = { factor = 1 }
    }
    option = {
        name = "We must endure."
        add_political_power = -50

        hidden_effect = { news_event = { id = GER_harvest.2 days = 14 } }
        ai_chance = { factor = 2 }
    }
}
news_event = {
    id = GER_harvest.2
    title = "‘Durchhalten!’"
    desc = GER_harvest.2.d

    is_triggered_only = yes

    option = {
        name = "If this is what it comes to..."
        if = {
            limit = { has_idea = idea_widespread_hunger }
            modify_timed_idea = { idea = idea_widespread_hunger months = 9 }
        }
        else = {
            add_timed_idea = { idea = idea_widespread_hunger months = 9 }
        }
    }
}
