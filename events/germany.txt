add_namespace = ger_wallonia

country_event = {
    id = ger_wallonia.1
    title = "The Occupation of Wallonia"
    desc = ger_wallonia.1.d

    is_triggered_only = yes

    option = {
        name = "Annex it."

        add_named_threat = { name = "The Annexation of Wallonia" threat = 4 }
        34 = { add_core_of = ROOT }

        hidden_effect = { 
            news_event = { id = ger_wallonia.2 hours = 6 }
            set_country_flag = imperial_tendency
        }
        ai_chance = { factor = 2 }
    }
    option = {
        name = "Keep the military administration."

        hidden_effect = { set_country_flag = neutral_tendency }
        ai_chance = { factor = 1 }
    }
    option = {
        name = "Withdraw our troops."

        add_popularity = { ideology = ROOT popularity = -0.05 }
        34 = { transfer_state_to = FRA }

        hidden_effect = { 
            news_event = { id = ger_wallonia.3 hours = 6 }
            set_country_flag = democratic_tendency 
        }
        ai_chance = { factor = 1 }
    }
}

# - atlantic
add_namespace = ger_atlantic

country_event = {
    id = ger_atlantic.1
    title = "Atlantic Submarine Warfare"
    desc = ger_atlantic.1.d

    is_triggered_only = yes
    trigger = { tag = GER has_war_with = ENG not = { has_country_flag = atlantic_event_done } }
    immediate = { hidden_effect = { set_country_flag = atlantic_event_done } }

    option = {
        name = "Gott Strafe England!"
        add_war_support = 0.05

        swap_ideas = {
            remove_idea = idea_tirpitz_plan
            add_idea = idea_atlantic_submarine_warfare
        }

        hidden_effect = {
            random = { chance = 50 USA = { news_event = { id = ger_atlantic.2 days = 28 random_days = 140 } } }
            every_country = {
                limit = { or = { tag = ARG tag = SPA } }
                random = { chance = 50 news_event = { id = ger_atlantic.3 days = 28 random_days = 140 } }
            }
        }

        ai_chance = { factor = 4 }
    }
    option = {
        name = "We will win this war in a conventional way."
        ai_chance = { factor = 1 }
    }
}

## -- 1916
add_namespace = GER_harvest

country_event = {
    id = GER_harvest.1
    title = "The Harvest of 1916"
    desc = GER_harvest.1.d

    trigger = {
        and = { tag = GER       date > 1916.7.1     date < 1916.10.1 }
        not = { has_idea = civilian_economy }
        not = { has_country_flag = harvest_event_done }
    }
    immediate = { hidden_effect = { set_country_flag = harvest_event_done } } 

    option = {
        name = "Demobilize the economy."
        custom_effect_tooltip = tt_demobilize_economy
        hidden_effect = {
            if = {
                limit = { has_idea = idea_rationing }
                remove_ideas = idea_rationing
            }
            add_timed_idea = { idea = idea_harvest_1916 months = 6 }
            add_ideas = civilian_economy
        }
        ai_chance = { factor = 1 }
    }
    option = {
        name = "We must endure."
        add_political_power = -50

        hidden_effect = { news_event = { id = GER_harvest.2 days = 14 } }
        ai_chance = { factor = 2 }
    }
}
news_event = {
    id = GER_harvest.2
    title = "‘Durchhalten!’"
    desc = GER_harvest.2.d

    is_triggered_only = yes

    option = {
        name = "If this is what it comes to..."
        if = {
            limit = { has_idea = idea_widespread_hunger }
            modify_timed_idea = { idea = idea_widespread_hunger months = 9 }
        }
        else = {
            add_timed_idea = { idea = idea_widespread_hunger months = 9 }
        }
    }
}
