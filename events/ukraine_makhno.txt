    id = ukraine_makhno.1
    title = "Anarchist Agitation in the South"
    desc = ukraine_makhno.1.d

    trigger = {
        tag = UKR
        any_state = {
            or = { state = 196 state = 200 }
            is_controlled_by = ROOT
        }
        or = {
            communism > 0.25
            SOV = { has_civil_war = yes }
        }
    }
    fire_only_once = yes

    option = {
        name = "Worrying."
        add_popularity = { ideology = communism popularity = 0.05 }
        add_stability = -0.05
    }
}
