add_namespace = japan_okhotsk

country_event = {
    id = japan_okhotsk.1
    title = "The Okhotsk Intervention"
    desc = japan_okhotsk.1.d

    trigger = { 
        tag = JAP
        SOV = { has_civil_war = yes }

        not = {
            any_country_with_original_tag = {
                original_tag_to_check = SOV
                has_war_with = JAP
            }
        }

        any_country_with_original_tag = {
            original_tag_to_check = SOV
            or = { controls_state = 408 controls_state = 409 controls_state = 655 }
        }
    }
    fire_only_once = yes

    option = {
        name = "This is the time to strike."

        every_state = {
            limit = {
                or = { state = 408 state = 409 state = 655 }
                controller = { original_tag = SOV }
            }
            transfer_state_to = JAP
        }

        every_country_with_original_tag = {
            original_tag_to_check = SOV
            create_wargoal = { target = JAP type = take_claimed_state }
        }

        news_event = { id = japan_okhotsk.2 hours = 6 }
    }
    option = {
        name = "No need to get involved."
        add_stability = 0.05
    }
}
