add_namespace = manchukuo_yuan

news_event = {
    id = manchukuo_yuan.1
    title = "Yuan Shikai Dies"
    desc = manchukuo_yuan.1.d

    trigger = {
        tag = MAN
        date > 1916.6.1
    }
    fire_only_once = yes

    option = {
        name = "This is a sad day."

        if = {
            limit = { has_country_leader = { character = MAN_yuan_shikai ruling_only = yes } }
            promote_character = MAN_tang_shaoyi
            hidden_effect = { news_event = { id = manchukuo_yuan.2 days = 60 } }
        }

        hidden_effect = { retire_character = MAN_yuan_shikai }
    }
}
