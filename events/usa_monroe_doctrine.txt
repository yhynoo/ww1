add_namespace = usa_monroe_doctrine

news_event = {
    id = usa_monroe_doctrine.1
    title = "Zapatistas Take Up Arms in Mexico!"                # Mexico is already fighting the Colombian invasion from last year, it's rough.
    desc = usa_monroe_doctrine.1.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "The peasants have had enough."

        if = {
            limit = { tag = MEX }
            start_civil_war = { ideology = communism size = 0.4 }

            hidden_effect = {
                add_popularity = { ideology = communism popularity = -0.2 }
                extra_troops_for_mexican_civil_war = yes

                USA = { country_event = { id = usa_monroe_doctrine.2 days = 7 random_hours = 6 } }
                COL = {
                    if = {
                        limit = { has_war_with = MEX }
                        every_country = {
                            limit = { original_tag = MEX not = { has_war_with = COL } }
                            declare_war_on = { target = PREV type = annex_everything }
                        }
                    }
                }
            }
        }
    }
}

country_event = {
    id = usa_monroe_doctrine.2
    title = "Aid to Mexican Government?"
    desc = usa_monroe_doctrine.2.d

    is_triggered_only = yes

    option = {
        name = "It is in our best interest."
        add_timed_idea = {
            idea = sww1_idea_aid_to_mexico
            years = 1
        }

        random_country = { 
            limit = { original_tag = MEX has_government = democratic }

            news_event = { id = usa_monroe_doctrine.3 hours = 12 random_hours = 6 }
            add_opinion_modifier = { target = USA modifier = medium_increase }
        }

        hidden_effect = {
            set_country_flag = usa_goes_to_mexico
        }

        ai_chance = { base = 1 }
    }

    option = {
        name = "This is not our war."
        add_popularity = { ideology = authoritarianism popularity = 0.05 }

        ai_chance = { base = 2 }
    }
}
