country_event = {
    id = poland_uprisings.2
    title = "Uprising in the West?"
    desc = poland_uprisings.2.d

    trigger = {
        original_tag = POL
        POL = { controls_state = 87 }       # we must have a border to make an uprising.
        GER = { 
            or = {
                has_civil_war = yes
                has_idea = idea_loser_of_the_great_war
            }
        }

        any_state = {
            or = { state = 85 state = 86 }
            is_owned_and_controlled_by = GER
        }
    }

    option = {
        name = "Let us see..."

        85 = { add_core_of = POL }
        86 = { 
            add_core_of = POL 
            save_event_target_as = uprising_target_state
        }

        GER = { country_event = { id = poland_uprisings.3 hours = 6 } }
    }
    option = {
        name = "This is not the time."
        add_political_power = -50
    }
}

country_event = {
    id = poland_uprisings.3
    title = "Armed Locals Gather in [uprising_target_state.GetName]!"
    desc = poland_uprisings.3.d

    is_triggered_only = yes

    option = {
        name = "We must withdraw."
        add_stability = 0.05

        every_owned_controlled_state = {
            limit = { is_core_of = POL }
            transfer_state_to = POL
        }
    }
    option = {
        name = "These are our lands!"

        hidden_effect = {
            if = {
                limit = { not = { POL = { has_template = "Uprising Troops" } } }
                POL = {
                    division_template = {
                        name = "Uprising Troops"
                        regiments = {
                            infantry = { x = 0 y = 0 }      infantry = { x = 0 y = 1 }
                            infantry = { x = 1 y = 0 }      infantry = { x = 1 y = 1 }
                        }
                    }
                }
            }
            event_target:uprising_target_state = {
                transfer_state_to = POL
                create_unit = {
                    division = "name = \"Uprising Troops\" division_template = \"Uprising Troops\" start_experience_factor = 0.5"
                    owner = POL
                    count = 3
                }
            }
        }

        declare_war_on = { target = FROM type = take_claimed_state }
    }
}
