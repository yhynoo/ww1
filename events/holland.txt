## -- neutrality
add_namespace = hol_neutrality

country_event = {
    id = hol_neutrality.1
    title = "The Dutch and the Great War"
    desc = hol_neutrality.1.d

    fire_only_once = yes
    mean_time_to_happen = { months = 1 }
    trigger = {
        tag = HOL
        FRA = { has_war_with = GER }
    }

    option = {
        name = "We shall stay neutral."
        add_ideas = idea_dutch_neutrality

        hidden_effect = { news_event = { id = hol_neutrality.2 hours = 6 } }
        ai_chance = { factor = 1 }
    }
    option = {
        name = "Let us keep our options open."
        add_war_support = 0.05
        add_popularity = { ideology = monarchism popularity = 0.025 }

        ai_chance = {
            factor = 0
            modifier = { has_completed_focus = HOL_adjust_the_government add = 4 }
        }
    }
}
