add_namespace = colombia_spain

country_event = {
    id = colombia_spain.1
    title = "[FROM.GetName] Offers Military Assistance"
    desc = colombia_spain.1.d 

    is_triggered_only = yes

    option = {
        name = "Their help is most welcome."
        FROM = { news_event = { id = colombia_spain.2 hours = 12 random_hours = 6 } }

        if = {
            limit = { is_in_faction = no }
            create_faction = "Hispanic Union"
        }
        add_to_faction = FROM

        hidden_effect = {
            FROM = {
                add_ai_strategy = {
                    type = alliance
                    id = ROOT
                    value = 200
                }
            }
        }

        ai_chance = { 
            base = 2
            modifier = { COL = { has_government = monarchism } factor = 3 }
        }
    }

    option = {
        name = "We are only interested in mutual defensive guarantees."
        trigger = {
            NOT = { 
                OR = { has_guaranteed = FROM FROM = { has_guaranteed = ROOT } }
            }
        }

        FROM = { country_event = { id = colombia_spain.3 hours = 12 random_hours = 6 } }
        ai_chance = { base = 1 }
    }

    option = {
        name = "We are not interested."
        FROM = { news_event = { id = colombia_spain.6 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 1
            modifier = { has_opinion = { target = FROM value < 0 } factor = 2 }
        }
    }
}
