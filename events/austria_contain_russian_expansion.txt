add_namespace = austria_contain_russian_expansion

### 1. Austria-Hungary Offers an Alliance to Afghanistan
country_event = {
    id = austria_contain_russian_expansion.1
    title = "Austria-Hungary Offers an Alliance to Afghanistan"
    desc = austria_contain_russian_expansion.1.d

    is_triggered_only = yes

    option = {
        name = "Accept the alliance offer."
        AFG = { country_event = { id = austria_contain_russian_expansion.2 hours = 12 random_hours = 6 } }

        if = {
            limit = { is_in_faction = no }
            create_faction = "Austro-Afghan Alliance"
        }
        faction_leader = { add_to_faction = AFG }
    }

    option = {
        name = "Reject the alliance offer."
        AFG = { country_event = { id = austria_contain_russian_expansion.3 hours = 12 random_hours = 6 } }
    }
}
