add_namespace = argentina_elections

country_event = {
    id = argentina_elections.1
    title = "The Election of 1916"
    desc = argentina_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = ARG
        has_government = democratic

        date > 1916.1.1         date < 1917.1.1
    }

    option = {
        name = "Hipólito Yrigoyen."
        add_popularity = { ideology = communism popularity = 0.025 }
        
        promote_character = ARG_hipolito_yrigoyen
        ai_chance = { 
            base = 1 
            modifier = { has_completed_focus = ARG_traditionalism factor = 0 }
        }
    }
    option = {
        name = "Lisandro de la Torre."

        add_war_support = 0.05
        promote_character = ARG_lisandro_de_la_torre

        ai_chance = { 
            base = 1 
            modifier = { has_completed_focus = ARG_adjust_the_government factor = 0 }
        }
    }
}
