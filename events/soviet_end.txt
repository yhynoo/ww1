add_namespace = soviet_end

country_event = {
    id = soviet_end.1
    title = "The End of the Empire?"
    desc = soviet_end.1.d

    fire_only_once = yes

    trigger = { 
        tag = SOV
        not = { has_government = communism }

        has_civil_war = no
        communism > 0.4
    }

    option = {
        name = "All power to the councils!"
        start_civil_war = { ideology = monarchism ruling_party = communism size = 0.5 }

        # manage the war
        hidden_effect = { russia_civil_war = yes }
    }
    option = {
        name = "We must hold out."
        start_civil_war = { ideology = communism size = 0.6 }

        # manage the war
        hidden_effect = { russia_civil_war = yes }
    }
}

news_event = {
    id = soviet_end.2
    title = "Uprising in the Baltics!"
    desc = soviet_end.2.d

    trigger = {
        tag = SOV
        has_civil_war = yes

        UBD = { exists = no }
        any_country_with_original_tag = {
            original_tag_to_check = SOV
            or = { controls_state = 12 controls_state = 13 }
        }
    }
    fire_only_once = yes

    option = {
        name = "We will take care of that."

        hidden_effect = { 
            baltic_independence = yes 
            UBD = { set_rule = { can_join_factions = no } }
        }
        every_country_with_original_tag = {
            original_tag_to_check = SOV
            declare_war_on = { target = UBD type = take_claimed_state }
        }
    }
}

            original_tag_to_check = SOV

    option = {
        name = "This will not last long."

        hidden_effect = {

        every_country_with_original_tag = {

        }
    }
}
