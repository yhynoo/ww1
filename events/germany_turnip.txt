add_namespace = germany_turnip

news_event = {
    id = germany_turnip.1
    title = "The Turnip Winter"
    desc = germany_turnip.1.d

    trigger = { 
        tag = GER
        has_war = yes 
        date > 1916.10.1
    }
    fire_only_once = yes

    option = {
        name = "That is not good."
        add_timed_idea = { idea = idea_turnip_winter months = 6 }

        hidden_effect = {
            country_event = { id = germany_turnip.2 days = 180 random_days = 30 }
        }
    }
}
