add_namespace = england_elections

country_event = {
    id = england_elections.1
    title = "The Election of 1914"
    desc = england_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = ENG
        has_government = democratic

        date > 1914.1.1         date < 1915.1.1
    }

    option = {
        name = "Herbert H. Asquith."
        add_stability = 0.05
        
        ai_chance = { base = 4 }
    }
    option = {
        name = "Arthur Balfour."

        add_popularity = { ideology = monarchism popularity = 0.025 }
        promote_character = ENG_arthur_balfour

        ai_chance = { base = 1 }
    }
}

country_event = {
    id = england_elections.2
    title = "The Election of 1918"
    desc = england_elections.2.d

    is_triggered_only = yes
    trigger = {
        tag = ENG
        has_government = democratic

        date > 1918.1.1         date < 1919.1.1
    }

    option = {
        name = "David Lloyd George."

        if = {
            limit = { has_country_leader = { character = ENG_david_lloyd_george ruling_only = yes } }
            add_stability = 0.05
        }
        else = {
            add_war_support = 0.05
            promote_character = ENG_david_lloyd_george
        }

        ai_chance = { 
            base = 2 
            modifier = { has_war = no factor = 0.5 }
        }
    }
    option = {
        name = "Arthur Balfour."
        
        if = {
            limit = { has_country_leader = { character = ENG_arthur_balfour ruling_only = yes } }
            add_stability = 0.05
        }
        else = {
            add_popularity = { ideology = monarchism popularity = 0.025 }
            promote_character = ENG_arthur_balfour
        }

        ai_chance = { base = 1 }
    }
}
