add_namespace = _armistice

country_event = {
    id = _armistice.1
    title = "Time to Seek an Armistice?"
    desc = _armistice.1.d

    is_triggered_only = yes
    trigger = { has_war = yes has_civil_war = no }

    option = {
        name = "Let us end this war."
        every_enemy_country = {
            limit = { 
                or = { is_faction_leader = yes is_in_faction = no } 
            }
            country_event = { id = _offer.1 hours = 6 random_days = 7 }
        }

        ai_chance = { 
            base = 2

            modifier = { 
                OR = { 
                    has_government = monarchism has_government = authoritarianism
                    has_war_support > 0.5
                }
                factor = 0
            }
        }
    }

    option = {
        name = "We shall continue fighting."
        add_political_power = -50

        ai_chance = { 
            base = 1 
            modifier = { surrender_progress > 0.1 factor = 0.5 }
        }
    }
}

    id = _armistice.2

    desc = _armistice.2.d

    }

        ai_chance = { base = 2 }

        ai_chance = { base = 1 }
    }
}


# anti-war demonstrations

news_event = {
    id = _armistice.3
    title = "Demonstrations Against War"
    desc = _armistice.3.d

    trigger = {
        has_war = yes
        or = { has_war_support < 0.3 has_stability < 0.3 }
        not = { has_idea = idea_anti_war_demonstrations }

        # war must be long enough
        longest_war_length > 18
    }

    mean_time_to_happen = { months = 1 }

    option = {
        name = "Worrying."
        add_timed_idea = { idea = idea_anti_war_demonstrations months = 6 }
    }
}
