# delegaion
add_namespace = gre_delegation

country_event = {
    id = gre_delegation.1
    title = "Cretan Delegation Arrives"   
    desc = gre_delegation.1.d

    is_triggered_only = yes
    trigger = { 182 = { not = { is_core_of = GRE } } }

    option = {
        name = "Welcome them."
        182 = { add_core_of = GRE }
        add_named_threat = { name = "The Annexation of Crete" threat = 4 }

        hidden_effect = { news_event = { id = gre_delegation.2 hours = 6 } }
        ai_chance = { factor = 1 }
    }
    option = {
        name = "Send them home."
        add_political_power = -25

        TUR = { add_opinion_modifier = { target = ROOT modifier = opinion_sent_delegation_home } }
        ai_chance = { factor = 0 }
    }
}
news_event = {
    id = gre_delegation.2
    title = "[GRE.GetName] Annexes Crete!"
    desc = gre_delegation.2.d

    is_triggered_only = yes
    major = yes

    option = {
        name = "The Turks will not like this..."
        if = {
            limit = { tag = TUR }
            add_opinion_modifier = { target = GRE modifier = opinion_annexed_crete }
        }
    }
}
