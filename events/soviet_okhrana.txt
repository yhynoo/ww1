    id = soviet_okhrana.2
    title = "Archduke Assassinated in Lemberg!"
    desc = soviet_okhrana.2.d

    is_triggered_only = yes
    trigger = { 
        AUS = { 
            controls_state = 91
            has_government = monarchism
        }
    }
    major = yes

    option = {
        name = "Horrifying."

        if = {
            limit = { tag = AUS } 
            
            add_stability = -0.1
            hidden_effect = { country_event = { id = soviet_okhrana.3 days = 7 random_days = 7 } }
        }
    }
}
