add_namespace = manchu

# xinhai
    country_event = {
        id = manchu.1
        title = "The Government Agreement"
        desc = manchu.1.d

        is_triggered_only = yes

        option = {
            name = "Honor the agreement."

            add_stability = 0.025
            promote_character = MAN_yuan_shikai
        }
        option = {
            name = "The army won't lead the country well."
            add_popularity = { ideology = democratic popularity = 0.05 }

            if = {
                limit = { or = { has_stability < 0.4 monarchism > 0.1 } }
                random = {
                    chance = 80
                    beiyang_war = yes
                }
            }
        }
    }

    country_event = {

        desc = manchu.2.d

        is_triggered_only = yes

        option = {
            name = "He is free to do that."
            release_on_controlled = TIB

            hidden_effect = { news_event = { id = manchu.3 hours = 6 } }
        }
        option = {
            name = "We must maintain the integrity of the state."
            TIB = {
                setup_uprising_country = yes
                save_event_target_as = uprising_country

                # flags

                add_timed_idea = { idea = idea_uprising months = 6 }
            }
            declare_war_on = { target = TIB type = annex_everything }
        }
    }

# warlords gain autonomy
    news_event = {
        id = manchu.4
        title = "Local Administrators Gain Autonomy"
        desc = manchu.4.d

        trigger = {
            has_idea = idea_warlordism_in_the_countryside       not = { has_country_flag = in_warlord_era }
            or = { has_stability < 0.2 has_civil_war = yes }
        }

        option = {
            name = "Worrying."
            add_popularity = { ideology = ROOT popularity = -0.05 }

            hidden_effect = { 
                set_country_flag = in_warlord_era
                set_warlord_cores = yes
            }
        }
    }

    news_event = {

            exists = yes

        immediate = {

            name = "We will not allow this."
