add_namespace = france_colombia

country_event = {
    id = france_colombia.1
    title = "[FROM.GetName] Offers Military Assistance"
    desc = france_colombia.1.d 

    is_triggered_only = yes

    option = {
        name = "That's a good idea."
        FROM = { 
            news_event = { id = france_colombia.2 hours = 12 random_hours = 6 }
            if = {
                limit = { is_in_faction = no }
                create_faction = "Versailles Accords"
            }
            faction_leader = { add_to_faction = ROOT }
        }

        hidden_effect = {
            add_ai_strategy = {
                type = alliance
                id = FROM
                value = 100
            }
        }

        ai_chance = { 
            base = 6
            modifier = { has_opinion = { target = FROM value > 50 } factor = 2 }
        }
    }

    option = {
        name = "We are only interested in mutual defensive guarantees."
        trigger = {
            NOT = { 
                OR = { has_guaranteed = FROM FROM = { has_guaranteed = ROOT } }
            }
        }

        FROM = { country_event = { id = france_colombia.3 hours = 12 random_hours = 6 } }
        ai_chance = { base = 1 }
    }

    option = {
        name = "We are not interested."
        FROM = { news_event = { id = france_colombia.6 hours = 12 random_hours = 6 } }

        ai_chance = { 
            base = 2
            modifier = { has_opinion = { target = FROM value < 0 } factor = 3 }
        }
    }
}
