    }
}

news_event = {
    id = soviet_civil_war.2
    title = "Nikolai II and His Family Executed?"
    desc = soviet_civil_war.2.d

    trigger = {
        original_tag = SOV
        has_government = communism

        controls_state = 195
        any_country = { has_country_leader = { character = SOV_nikolai_ii ruling_only = yes } }
    }

    major = yes

    option = {
        trigger = { original_tag = SOV has_government = communism }
        name = "Down with the tyrant!"
        add_political_power = 100
    }
    option = {
        trigger = { original_tag = SOV not = { has_government = communism } }
        name = "God save his soul."

        retire_character = SOV_nikolai_ii
        add_war_support = -0.2
    }
    option = {
        trigger = { not = { original_tag = SOV } }
        name = "Outrageous."
    }
}

# independence wars
news_event = {
    id = soviet_civil_war.3
    title = "Poland Emerges Out of Empire's Ruins!"
    desc = soviet_civil_war.3.d

    is_triggered_only = yes
    major = yes

    trigger = {
        POL = { exists = no }
        any_country = { original_tag = SOV or = { controls_state = 10 controls_state = 87 controls_state = 90 controls_state = 97 } }
    }

    option = {
        name = "For our freedom and yours!"
        
        if = {
            limit = { tag = SOV }

            hidden_effect = { 
                start_polish_independence_war = yes
                POL = { 
                    set_politics = { ruling_party = democratic elections_allowed = yes } 
                    set_popularities = { communism = 10 democratic = 55 monarchism = 35 }
                    set_rule = { can_join_factions = no }
                }
            }
        }
    }
}
