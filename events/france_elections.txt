add_namespace = france_elections

country_event = {
    id = france_elections.1
    title = "The Election of 1913"
    desc = france_elections.1.d

    is_triggered_only = yes
    trigger = {
        tag = FRA
        has_government = democratic

        date > 1913.1.1         date < 1914.1.1
    }

    option = {
        name = "Louis Barthou."

        add_popularity = { ideology = monarchism popularity = 0.025 }
        promote_character = FRA_louis_barthou

        ai_chance = { base = 1 }
    }
    option = {
        name = "Aristide Briand."
        add_stability = 0.05

        ai_chance = { base = 2 }
    }
}

    title = "The Election of 1921"
    desc = france_elections.3.d

    is_triggered_only = yes
    trigger = {
        tag = FRA
        has_government = democratic

        date > 1921.1.1         date < 1922.1.1
    }

    option = {
        name = "Georges Clemenceau."

        if = {
            limit = { has_country_leader = { character = FRA_georges_clemenceau ruling_only = yes } }
            add_stability = 0.05
        }
        else = {
            add_war_support = 0.05
            promote_character = FRA_georges_clemenceau
        }
        
        ai_chance = { 
            base = 2
            modifier = { has_war = yes factor = 0 }     # enough of this war...
        }
    }
    option = {
        name = "Édouard Herriot."

        add_popularity = { ideology = democratic popularity = 0.025 }
        promote_character = FRA_edouard_herriot

        ai_chance = { base = 1 }
    }
