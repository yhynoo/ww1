### 1. Germany Proposes the Kaiserliche Eisenbahn
country_event = {
    id = germany_oriental.1
    title = "German Proposal: The Kaiserliche Eisenbahn"
    desc = germany_oriental.1.d

    is_triggered_only = yes

    option = {
        name = "Accept the German railway investment."
        GER = { news_event = { id = germany_oriental.2 hours = 12 random_hours = 6 } }

        hidden_effect = {
            country_event = { id = germany_oriental.4 days = 2 random_hours = 6 }
            107 = {
                add_building_construction = {
                    type = infrastructure
                    level = 1
                    instant_build = yes
                }
            }
            737 = {
                add_building_construction = {
                    type = infrastructure
                    level = 1
                    instant_build = yes
                }
            }
        }

        ai_chance = { base = 6 }
    }

    option = {
        name = "Deny German involvement in our railways."
        GER = { news_event = { id = germany_oriental.3 hours = 12 random_hours = 6 } }

        ai_chance = { base = 1 }

        if = {
            limit = { not = { has_country_flag = we_do_our_thing } }
            set_country_flag = we_do_our_thing
        }
    }
}

    title = "Shall We Request German Guarantees?"
    desc = germany_oriental.4.d

    is_triggered_only = yes

    # if we decided to do our thing, we don't do this.
    trigger = { not = { has_completed_focus = TUR_sarajevo_memorandum } }

    option = {
        name = "Yes, we need Germany's protection."
        GER = { country_event = { id = germany_oriental.5 hours = 24 random_hours = 12 } }

        ai_chance = { 
            base = 6
            modifier = { has_country_flag = we_do_our_thing factor = 0 }
        }
    }

    option = {
        name = "No, we can stand on our own."
        ai_chance = { base = 1 }

        if = {
            limit = { not = { has_country_flag = we_do_our_thing } }
            set_country_flag = we_do_our_thing
        }
    }
}
